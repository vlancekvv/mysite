import React, { Component } from 'react';
import { View, StyleSheet, ActivityIndicator, Platform, Text, Dimensions, AppState } from 'react-native';
import { connect } from 'react-redux';
import DeviceInfo from 'react-native-device-info';
import SplashScreen from 'react-native-splash-screen';
import BackgroundFetch from 'react-native-background-fetch';
 
import { Root, Login } from './services/Router';
import { colors, layouts} from './shared/styles';
import { setOrientation, enableTabletView, changeAppState, getLiveServer, triggerLoader } from './actions/index';
import { getUpdatedOrders, handleSynchronizationError, setOrders, setUser } from './actions/backend';
import StorageService from './services/StorageService';
import Config from './utils/config';
import AppInfo from './views/AppInfo';
 
class AppView extends Component {
 
  constructor(props){
    super(props);
    this.state = {
      auth: true
    };
    this.startBackgroundUpdating = this.startBackgroundUpdating.bind(this);
  }
 
  componentWillMount = () => {
    this.tablet();
  };
 
  componentDidMount() {
    SplashScreen.hide();
    this.checkAuth();
    this.props.getLiveServer();
  }
 
  componentWillReceiveProps = (nextProps) => {
    if (nextProps.authorized !== this.props.authorized) {
      this.checkAuth();
    }
    if (this.props.isManualUpdating) {
      clearInterval(this.updatingInterval);
      this.startAutoOrderUpdating(Config.UPDATING_TIMER_FOREGROUND);
    }
  };
 
  componentWillUnmount() {
    AppState.removeEventListener('change', this.prepareAutoOrderUpdating);
    clearInterval(this.updatingInterval);
  }
 
  startBackgroundUpdating() {
    BackgroundFetch.configure({
      minimumFetchInterval: Config.UPDATING_TIMER_BACKGROUND,
    }, () => {
      this.props.getUpdatedOrders()
        .catch(error => this.props.handleSynchronizationError({...error, message: 'Verbindungsproblem…'}))
        .then(() => BackgroundFetch.finish(BackgroundFetch.FETCH_RESULT_NEW_DATA));
    });
  }
 
  prepareAutoOrderUpdating = (state) => {
    this.props.changeAppState(state);
    clearInterval(this.updatingInterval);
    if (state === 'active') {
      this.startAutoOrderUpdating(Config.UPDATING_TIMER_FOREGROUND);
    }
  };
 
  startAutoOrderUpdating = (timer) => {
    this.updatingInterval = setInterval(() => this.props.getUpdatedOrders()
      .catch(error => this.props.handleSynchronizationError({...error, message: 'Verbindungsproblem…'})), timer)
  };
 
  checkAuth = () => {
    StorageService.getAuthorization().then(i => {
      if (i === undefined || i.state === false ) {
        BackgroundFetch.stop();
        this.setState({auth: false});
        clearInterval(this.updatingInterval);
        AppState.removeEventListener('change', this.prepareAutoOrderUpdating);
      } else {
        this.startBackgroundUpdating();
        this.setState({auth: true});
        this.props.setUser(i.mail);
        AppState.addEventListener('change', this.prepareAutoOrderUpdating);
        this.loadOrders()
          .then(() => this.prepareAutoOrderUpdating(this.props.appState));
      }
    })
  };
 
  setOrientationApp = (event) => {
    const width = event.nativeEvent.layout.width;
    const height =  event.nativeEvent.layout.height;
    if (width > height){
      this.props.setOrientation('LANDSCAPE')
    }
    else {
      this.props.setOrientation('PORTRAIT')
    }
  };
 
  tablet = () => {
    const isTablet = DeviceInfo.isTablet();
    if ( isTablet === true ) {
      this.props.enableTabletView()
    }
  };
 
  loadOrders = () => {
    this.props.triggerLoader();
    return StorageService.getOrders()
      .then((body) => {
        this.props.setOrders(body ? body.orders : []);
        this.props.triggerLoader();
      })
  };
 
  render() {
    const prefix = Platform.OS === 'android' ? 'immoinspect://immoinspect/' : 'immoinspect://';
    return(
      <View style={[ layouts.fullOverlay ]}>
        { this.props.loading ?
          <View style={[ layouts.fullOverlay, styles.loaderBackground ]}>
            <ActivityIndicator size="large" color="#0000ff"/>
            {this.props.progress ?
              <Text style={styles.progressText}>
                {
                  this.props.progress < 100 ?
                  `Übertragen des Auftrags... ${this.props.progress}%`
                  : 'Warten auf Bestätigung vom Server...'
                }
              </Text> : null
            }
          </View> : null
        }
         <View style={[ layouts.fullOverlay, this.props.loading ? {flex: 0, opacity: 0} : null ]}
               onLayout={(event) => this.setOrientationApp(event)}>
           {(this.state.auth) ? <Root uriPrefix={prefix}/> : <Login/>}
           <AppInfo />
        </View>
      </View>
    )
  }
}
 
const mapStateToProps = state => {
  return {
    loading: state.App.loading,
    authorized: state.Api.authorized,
    progress: state.Api.progressLoading,
    isManualUpdating: state.Api.isManualUpdating,
    appState: state.App.appState
  }
};
 
const mapDispatchToProps = dispatch => {
  return {
    enableTabletView: () => dispatch(enableTabletView()),
    setOrientation: (orientation) => dispatch(setOrientation(orientation)),
    getUpdatedOrders: (flag) => dispatch(getUpdatedOrders(flag)),
    setOrders: (orders) => dispatch(setOrders(orders)),
    triggerLoader: () => dispatch(triggerLoader()),
    changeAppState: (state) => dispatch(changeAppState(state)),
    setUser: (userName) => dispatch(setUser(userName)),
    handleSynchronizationError: error => dispatch(handleSynchronizationError(error)),
    getLiveServer: () => dispatch(getLiveServer())
  }
};
 
const styles = StyleSheet.create({
  loaderBackground: {
    position: 'absolute',
    top: 0,
    left: 0,
    backgroundColor: colors.white,
    zIndex: 10,
    alignItems: 'center',
    justifyContent: 'center',
    width: Dimensions.get('window').width,
    height: Dimensions.get('window').height
  },
  progressText: {
    marginTop: 20
  }
});
 
export default connect(mapStateToProps, mapDispatchToProps)(AppView)
