	static getItems() {
		return AsyncStorage.getAllKeys()
			.then((keys) => keys.filter((i) => {
				return i.startsWith('item_');
			}))
			.then((keys) => AsyncStorage.multiGet(keys))
			.then((stores) => {
				return stores.map((result) => {
					return JSON.parse(result[1]);
				});
			});
	}

	static upsertItem(item) {
		if (!item.id) {
			item.id = this.createID();
		}
		return this.getItem(item.id).then((i) => {
			if (i) {
				return AsyncStorage.getItem("item_" + item.id, JSON.stringify(item));
			} else {
				return this.saveItem(item);
			}
		});
	}

  static getOrder(id) {
    return db.get('order')
      .then((data) => {
        let neededOrder = false;
        data.orders.map((order) => {
          if (id === order.goVID) neededOrder = order
        });
        if (!neededOrder) throw new Error("No order found");
        return neededOrder
      })
      .catch((err) => {
        console.log(err.message);
      });
  }

	static createID() {
		const gen = (count) => {
			let out = '';
			for (let index = 0; index < count; index++) {
				out += (((1 + Math.random()) * 0x10000) | 0).toString(16).substring(1)
			}
			return out
		};
		return [gen(2), gen(1), gen(1), gen(1), gen(3)].join('-')
	}
