import base64 from "base-64";
import config from "../utils/config";
import axios from 'axios';
import { NetInfo } from 'react-native';
import global from "../shared/global";
 
const CancelToken = axios.CancelToken;
 
export default class ApiService {
 
  static getHandler(cancelRequestCallback) {
    return (connection) => connection.type === 'none' && cancelRequestCallback();
  };
 
  static getOrders(auth) {
    const source = CancelToken.source();
    const handler = this.getHandler(() => source.cancel());
    NetInfo.addEventListener('connectionChange', handler);
    return axios({
      method: 'GET',
      url: `${config[`${global.useLiveServer}_API_BASE`]}/webservice/app/v3/service/inspections/orders?otherStatuses=true`,
      headers: {
        'Content-Type': 'multipart/form-data',
        "Authorization": `Basic ${base64.encode(`${auth.mail}:${auth.password}`)}`
      },
      cancelToken: source.token,
      timeout: 60000
    }).then((resp) => {
      NetInfo.removeEventListener('connectionChange', handler);
      return resp.data.orders || []
    });
  }
 
  static getUpdatedOrders(auth, updatedSince) {
    return axios({
      method: 'GET',
      url: `${config[`${global.useLiveServer}_API_BASE`]}/webservice/app/v3/service/inspections/orders?otherStatuses=true&updatedSince=${updatedSince}`,
      headers: {
        'Content-Type': 'multipart/form-data',
        "Authorization": `Basic ${base64.encode(`${auth.mail}:${auth.password}`)}`
      },
      timeout: 180000
    }).then((resp) => resp.data.orders || []);
  }
 
  static saveOrder(auth, orderId, data, onProgress) {
    return new Promise((resolve, reject) => {
      const url = `${config[`${global.useLiveServer}_API_BASE`]}/webservice/app/v3/service/inspections/values/${orderId}?closing=false&content=photos&validate=true`;
      const xhr = new XMLHttpRequest();
      const handler = ApiService.getHandler(() => {
        xhr.abort();
        reject(new Error('Kein Netzwerk'));
      });
      NetInfo.addEventListener('connectionChange', handler);
      xhr.open('PUT', url, true);
      xhr.setRequestHeader('Content-Type', 'multipart/form-data');
      xhr.setRequestHeader('Authorization', `Basic ${base64.encode(`${auth.mail}:${auth.password}`)}`);
      if (xhr.upload) {
        xhr.upload.onprogress = onProgress;
      }
      xhr.onerror = (error) => {
        NetInfo.removeEventListener('connectionChange', handler);
        reject({
          ...error,
          code: xhr.status
        })
      };
      xhr.onload = () => {
        NetInfo.removeEventListener('connectionChange', handler);
        resolve();
      };
      xhr.send(JSON.stringify({values: data}));
    });
  }
 
  static login(userName, userPassword) {
    return axios({
      method: 'GET',
      url: `${config[`${global.useLiveServer}_API_BASE`]}/webservice/app/v3/service/inspections/login`,
      headers: {
        'Content-Type': 'multipart/form-data',
        "Authorization": `Basic ${base64.encode(`${userName}:${userPassword}`)}`
      }
    });
  }
}
