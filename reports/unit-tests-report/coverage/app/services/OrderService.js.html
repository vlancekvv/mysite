  static _getData(photosMap) {
    let { dataCategory } = store.getState().App;
    dataCategory = dataCategory.filter(category => !category.hasOwnProperty('state'));
    let values = [];
    dataCategory.forEach((category) => {
      let count = 0, key = category.name;
      if (key !== 'Außenfotos' && key !== 'Schadenfotos Außen' && key !== 'Schadenfotos Innen') {
        values.push({
          value: category.visible,
          key: database_keys[key].bool_key
        });
      }
      if (category.visible) {
        photosMap[key].forEach(photo => {
          if (photo.category_key === 'opt') {
            count++;
            values.push(this._getValue(database_keys[photo.category].optional + count, photo));
          } else {
            values.push(this._getValue(database_keys[photo.category].require + photo.category_key, photo));
          }
        });
      }
    });

  }
