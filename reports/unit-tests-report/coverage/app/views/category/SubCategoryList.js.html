import React, { Component } from 'react'
import { View, StyleSheet, Dimensions, Image, Text, TouchableOpacity, Alert } from 'react-native'
import { connect } from 'react-redux'
import { setCurrentPhoto, setNeededCategory, toggleAssignPhotoModal, reorderPhoto } from '../../actions'
import StorageService from '../../services/StorageService';
 
class SubCategoryList extends Component {
 
   setPhoto = () => {
     this.props.setNeededCategory(this.props.item.category, this.props.item);
     this.props.setCurrentPhoto(this.props.item);
     this.props.toggleModal();
   };
 
   reorderPhoto = () => {
     StorageService.saveItem({...this.props.item, category_key: this.props.currentPhoto.category_key})
       .then(() => StorageService.saveItem({...this.props.currentPhoto, category_key: this.props.item.category_key}))
       .then(() => this.props.reorderPhoto(this.props.item))
       .catch(error => Alert.alert(error.message))
       .then(() => this.props.toggleAssignPhotoModal());
   };
 
  render() {
    const item = this.props.item;
    const tablet = this.props.tablet;
    return (
    <TouchableOpacity disabled={this.props.disable || (this.props.isReorder && this.props.currentPhoto.id === item.id)}
                      style={!tablet ? styles.container : styles.containerTablet}
                      onPress={() => this.props.isReorder ? this.reorderPhoto() : this.setPhoto()}>
        <View style={styles.imageWrap}>
            <Image resizeMode='contain'
                   style={styles.image}
                   source={{uri: this.props.item.img.thumbnail}}/>
        </View>
        <View style={styles.detailsContainer}>
          <Text style={styles.bold}>{!this.props.item.description.length ? this.props.item.category : this.props.item.description}</Text>
          <Text>{this.props.item.date}</Text>
        </View>
        <View style={styles.detailsContainerRight}>
            <Text>{this.props.index === 'opt' ? 'opt.' : `#${this.props.index}`}</Text>
        </View>
      </TouchableOpacity>
    )
  }
}
const styles = StyleSheet.create({
  container: {
    flexDirection: 'row',
    alignItems: 'center',
    borderBottomColor: 'gray',
    borderBottomWidth: 1,
    marginRight: 10,
    justifyContent:'space-between',
  },
  containerTablet:{
    flexDirection: 'row',
    alignItems: 'center',
    borderBottomColor: 'gray',
    borderBottomWidth: 1,
    marginRight: 10,
    justifyContent:'space-between',
  },
  image: {
    flex: 1,
    alignSelf: 'stretch'
  },
  detailsContainer: {
    flex:1,
    flexDirection: 'column',

    alignItems: 'flex-start',
    paddingHorizontal:20,
  },
  imageWrap: {
    backgroundColor: '#fff',
    maxWidth: 85,
    maxHeight: 85,
    width: (Dimensions.get('window').width / 7.5),
    height:(Dimensions.get('window').width / 7.5),
    flexWrap: 'wrap',
    flexDirection: 'column',
    alignSelf: 'flex-start'
  },
  bold:{
    fontWeight: 'bold',
    paddingVertical: 5,
  },
  detailsContainerRight: {
    flexDirection: 'column',
    alignItems: 'flex-end',
    paddingHorizontal:20,
  },
});
 
const mapStateToProps = state => {
  return {
    tablet: state.App.isTablet,
    currentPhoto: state.App.currentPhotoItem
  }
};
 
const mapDispatchToProps = dispatch => {
  return {
    setCurrentPhoto: (item) => dispatch(setCurrentPhoto(item)),
    setNeededCategory: (name,item) => dispatch(setNeededCategory(name,item)),
    toggleAssignPhotoModal: () => dispatch(toggleAssignPhotoModal()),
    reorderPhoto: photo => dispatch(reorderPhoto(photo))
  }
};
 
export default connect(mapStateToProps, mapDispatchToProps)(SubCategoryList)
