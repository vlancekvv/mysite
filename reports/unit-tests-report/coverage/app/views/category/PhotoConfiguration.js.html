import React, { Component } from 'react'
import { View, StyleSheet, Dimensions, Text, TouchableOpacity, TextInput, ScrollView, Alert } from 'react-native'
import StorageService from '../../services/StorageService'
import { connect } from 'react-redux'
import {
  changeCurrentPhoto,
  setCategoryKey,
  setCurrentPhoto,
  setNeededCategory,
  toggleAssignPhotoModal,
  toggleDescription,
} from '../../actions'
import { ActionSheetCustom as ActionSheet } from 'react-native-actionsheet'
import Modal from "react-native-modal"
import InspectionModels from "../../shared/InspectionModels";
 
class PhotoConfiguration extends Component {
 
  constructor(props){
    super(props);
    this.state = {
      active: false,
      modal:false,
      isModalVisible: false,
      isDescriptionModalVisible: false,
      isSlotModal: false,
      reload: false,
      rectangle: false,
      category: InspectionModels[this.props.modelName].attributes.filter(attr => attr.fieldType === 'CONTAINER_PHOTOS')
        .map(attr => attr.name),
      description: [],
      renderActionCategory: false,
      descrLegth: '',
      currentDescription: '',
      isReorder: false,
      isAssignMode: true
    }
  }
 
  componentDidUpdate(prevProps) {
    if (this.props.itemPhoto !== prevProps.itemPhoto) {
      this.props.attributes.map(attr => {
        if (this.props.itemPhoto.category === attr.name) {
          this.setState({
            description: attr.properties.choices,
            descrLegth: attr.properties.choices.length,
            currentDescription: this.props.itemPhoto.description
          });
        }
      })
    }
  }
 
  _renderSlotModal = () => {
    const {minPhotos, maxPhotos} = this.props.attributes.find(attr => attr.name === this.props.neededCategory).properties;
    const optPhotos = this.props.photos.filter(photo => photo.category === this.props.neededCategory && photo.category_key === 'opt');
    return (
      <Modal animationInTiming={10} ref="myRef" isVisible={this.state.isSlotModal && this.props.isShowAssignPhotoModal}>
        <View style={styles.modalContentSlot}>
          {minPhotos ? <ScrollView style={styles.modalSlot}>
            {this.props.renderSlot(this.state.isReorder)}
          </ScrollView> : null}
          {maxPhotos - minPhotos > optPhotos.length ?
            <TouchableOpacity style={styles.modalClick}
                              onPress={() => {
                                this.setState({isSlotModal: false});
                                !this.state.isReorder ?
                                  this.props.setCategoryKey('opt') :
                                  this.props.assignPhotoToCategory(this.props.itemPhoto.description, this.props.itemPhoto, 'opt')
                              }}>
              <Text style={styles.modalText}>Als optionales Foto verwenden</Text>
            </TouchableOpacity> : null}
          <TouchableOpacity style={styles.cancelModalClick}
                            onPress={() => {
                              this.setState({isSlotModal: false});
                              this.props.toggleAssignPhotoModal();
                            }}>
            <Text style={styles.modalText}>Abbrechen</Text>
          </TouchableOpacity>
        </View>
      </Modal>
    )
  };
 
 
  renderPreDescription = () => {
    return(
      <Modal animationInTiming={10}
             ref="myRef"
             isVisible={this.state.isModalVisible}
             style={{justifyContent:'center'}}>
        <View style={styles.modalContent}>
          <Text style={{textAlign:'center',fontSize:25, paddingVertical:10}}>Bildbetitelung bearbeiten</Text>
          <Text style={{textAlign:'center',fontSize:17, paddingVertical:10}}>
            Sie können den Textbaustein bearbeiten oder einen anderen auswählen.
          </Text>
          <TextInput style={styles.modalInput}
                     underlineColorAndroid = "transparent"
                     placeholderTextColor = "#9a73ef"
                     multiline
                     autoCapitalize = "none"
                     value = {this.state.currentDescription}
                     onChangeText = {this.handleText}/>
          <TouchableOpacity
            style={styles.modalClick}
            onPress={this.assignNewDescription}>
            <Text style={styles.modalText}>Bestätigen</Text>
          </TouchableOpacity>
          <TouchableOpacity style={styles.modalClick}
                            onPress={() => this.setState({isDescriptionModalVisible: true, isModalVisible: false})}>
            <Text style={styles.modalText}>Anderer Textbaustein</Text>
          </TouchableOpacity>
          <TouchableOpacity style={styles.modalClick}
                            onPress={() => this.setState({isModalVisible: false})}>
            <Text style={styles.modalText}>Abbrechen</Text>
          </TouchableOpacity>
        </View>
      </Modal>
    )
  };
 
  renderDescriptionSelect = () => {
    return (
      <Modal isVisible={this.state.isDescriptionModalVisible || this.props.isShowDescriptionModal} style={{justifyContent:'center'}}>
        <View style={styles.modalContentDescription}>
          <ScrollView style={{width: '100%', marginTop: 10}} contentInset={{top: 10}}>
            {
              this.props.attributes.find(attr => attr.name === this.props.neededCategory).properties.choices
                .map((name, key) =>
                  <TouchableOpacity key={key}
                                    style={{...styles.modalClick, backgroundColor: '#f4f4f4', borderBottomColor: '#c3c3c3'}}
                                    onPress={() => this.selectDescription(name)}>
                    <Text style={{...styles.modalText, color: 'black', fontSize: 16}}>{name}</Text>
                  </TouchableOpacity>
                )
            }
          </ScrollView>
            <TouchableOpacity style={{...styles.cancelModalClick, borderTopColor: '#bdbdbd', borderTopWidth: 0.5}}
                              onPress={() => {
                                this.props.toggleDescription(false);
                                this.setState({isDescriptionModalVisible: false, isModalVisible: !this.state.isAssignMode})
                              }}>
              <Text style={styles.modalText}>Abbrechen</Text>
            </TouchableOpacity>
        </View>
      </Modal>
    );
  };
 
  toPostCapture = (r) => {
    this.props.changeCurrentPhoto(r.id);
    StorageService.getOriginal(r.id).then((original) => {
      const payload = { ...r };
      payload.img = { ...payload.img, original };
      this.props.navigate.navigate('PostCaptureView', {
        payload,
        title: this.props.orderName,
        id: this.props.order,
        databaseId: r.id,
        isAddedPhoto: true
      })
    })
      .catch(() => Alert.alert('Cannot get photo from storage'))
  };
 
  assignNewDescription = () => {
    if (this.state.currentDescription.length) {
      this.props.handleAssignDescription(this.props.itemPhoto, this.state.currentDescription);
      this.setState({isModalVisible: false, currentDescription: ''})
    } else {
      Alert.alert('Bildbetitelung darf nicht leer sein');
    }
  };
 
  selectDescription = (name) => {
    if (this.state.isAssignMode) {
      this.props.assignPhotoToCategory(name, this.props.itemPhoto);
      this.setState({currentDescription: '', isDescriptionModalVisible: false})
    } else {
      this.setState({currentDescription: name, isDescriptionModalVisible: false, isModalVisible: true});
    }
    this.props.toggleDescription(false);
  };
 
  assignCategory = (key) => {
    const neededAttribute = this.props.attributes[key];
    if (neededAttribute) {
      this.props.setNeededCategory(neededAttribute.name, this.props.itemPhoto);
      this.props.toggleAssignPhotoModal();
      this.setState({isSlotModal: true});
    }
  };
 
  handleReload = () => {
    this.setState({reload: !this.state.reload})
  };
 
  handleText = (text) => {
    this.setState({ currentDescription: text})
  };
 
  reorderPhoto = () => {
    this.props.toggleModal();
    this.setState({isReorder: true});
    setTimeout(() => {
      this.setState({isSlotModal: true});
      this.props.toggleAssignPhotoModal();
    }, 40)
  };
 
  render() {
    const category = this.props.itemPhoto.category;
    return (
      <View>
        <ActionSheet ref={o => this.ActionSheet = o}
                     title={'Kategorie'}
                     options={[...this.state.category, 'Abbrechen']}
                     cancelButtonIndex={this.state.category.length}
                     onPress={(index) => this.assignCategory(index)}
        />
        {(this.state.isDescriptionModalVisible || this.props.isShowDescriptionModal) && this.renderDescriptionSelect()}
        {this.state.isModalVisible && this.renderPreDescription()}
        {this.props.isShowAssignPhotoModal && this._renderSlotModal()}
        <Modal animationOutTiming={10} key={0} isVisible={this.props.showModal} style={{justifyContent: 'flex-end'}}>
          <View style={styles.modalContent}>
            <TouchableOpacity style={styles.modalClick}
                              onPress={() => {
                                this.props.toggleModal();
                                setTimeout(() => {
                                  this.ActionSheet.show();
                                  this.setState({isAssignMode: true});
                                }, 40)
                              }}>
              <Text style={styles.modalText}>Foto zuordnen</Text>
            </TouchableOpacity>
            { category ? <TouchableOpacity style={styles.modalClick}
                                           onPress={() => this.reorderPhoto()}>
              <Text style={styles.modalText}>Foto verschieben</Text>
            </TouchableOpacity> : null }
            { category ? <TouchableOpacity style={styles.modalClick}
                                           onPress={() => {
                                             this.props.toggleModal();
                                             setTimeout(() => this.setState({isModalVisible: true, isAssignMode: false}),40);
                                           }}>
              <Text style={styles.modalText}>Bildbetitelung bearbeiten</Text>
            </TouchableOpacity> : null }
            { category ? <TouchableOpacity style={styles.modalClick}
                                           onPress={() => {
                                             this.props.deAssignCategory(this.props.itemPhoto);
                                             this.props.toggleModal();
                                           }}>
              <Text style={styles.modalText}>Fotozuor‌d‌nung trennen</Text>
            </TouchableOpacity> : null }
            <TouchableOpacity style={styles.modalClick}
                              onPress={() => {
                                this.toPostCapture(this.props.itemPhoto);
                                this.props.toggleModal();
                              }}>
              <Text style={styles.modalText}>Foto bearbeiten</Text>
            </TouchableOpacity>
            <TouchableOpacity style={styles.modalClick}
                              onPress={() => this.props.deletePhoto(this.props.itemPhoto)}>
              <Text style={styles.deleteModalText}>Foto löschen</Text>
            </TouchableOpacity>
            <TouchableOpacity style={styles.cancelModalClick}
                              onPress={() => this.props.toggleModal()}>
              <Text style={styles.modalText}>Abbrechen</Text>
            </TouchableOpacity>
          </View>
        </Modal>
      </View>
    )
  }
}

const mapStateToProps = state => {
  return {
    reload: state.Home.reloadGallery,
    total:state.Home.totalFile,
    tablet: state.App.isTablet,
    modelName: state.App.modelName,
    orderName: state.App.currentOrderName,
    order: state.App.currentOrder,
    photos: state.App.dataPhotos,
    neededCategory: state.App.neededCategory,
    isShowAssignPhotoModal: state.App.isShowAssignPhotoModal,
    itemPhoto: state.App.currentPhotoItem,
    attributes: state.App.attributes,
    isShowDescriptionModal: state.App.isShowDescriptionModal
  }
};
 
const mapDispatchToProps = dispatch => {
  return {
    changeCurrentPhoto: (id) => dispatch(changeCurrentPhoto(id)),
    setCurrentPhoto: (item) => dispatch(setCurrentPhoto(item)),
    setNeededCategory: (name,item) => dispatch(setNeededCategory(name,item)),
    toggleAssignPhotoModal: () => dispatch(toggleAssignPhotoModal()),
    toggleDescription: (flag) => dispatch(toggleDescription(flag)),
    setCategoryKey: key => dispatch(setCategoryKey(key))
  }
};
 
export default connect(mapStateToProps, mapDispatchToProps)(PhotoConfiguration)
