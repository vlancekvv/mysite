import React, { Component } from "react"
import {View, StyleSheet, ScrollView, Text, Alert, BackHandler,} from "react-native"
import { connect } from "react-redux"
import CategoryList from "./CategoryList"
import { Switch } from 'react-native-switch'
import StorageService from '../../services/StorageService'
import {
  addCategoryData,
  addPhotos, assignCategory, assignDescription,
  changeCurrentOrder,
  changeCurrentOrderName, checkTotal, deAssignCategory,
  deAssignPhotoCategory, deleteItem, deletePhoto, handleCategorySwitch,
  resetCategoryData, setReloadingCategoriesFlag, triggerLoader,
  validateCategory
} from "../../actions";
import ModalMap from "../MapModal";
 
class ExecutionCategoryView extends Component {
 
  constructor(props) {
    super(props);
    this.state = {
      isShowCategories: false
    }
  }
 
  componentDidMount = () => {
    this.props.setReloadingCategoriesFlag(false);
    BackHandler.addEventListener('hardwareBackPress', this.handleBackPress);
    this.handleUrl();
  };
 
  componentWillReceiveProps = (nextProps) => {
    if (nextProps.order && (nextProps.isOrderValid !== this.props.isOrderValid)) {
      return StorageService.getItem(nextProps.order).then(item => {
        if (item) {
          item.valid = nextProps.isOrderValid;
          return StorageService.saveItem(item);
        }
      });
    }
  };
 
  componentDidUpdate(prevProps) {
    if (this.props.isReloadingCategories && this.props.appStateWasChanged) {
      this.props.setReloadingCategoriesFlag(false);
      this.handleUrl();
    }
    if (prevProps.order !== this.props.order) {
      this.loadItems();
    }
  }
 
  componentWillUnmount() {
    BackHandler.removeEventListener('hardwareBackPress', this.handleBackPress);
  }
 
  handleBackPress = () => {
    this.props.navigation.navigate('Home');
    return true;
  };
 
  handleUrl = () =>  {
    let urlId = +this.props.navigation.state.params.id;
    if (urlId) {
      const currentOrder = this.props.orders.find(order => order.goVID === urlId);
      if (currentOrder) {
        this.props.changeCurrentOrder(currentOrder);
        const title = `${currentOrder.street} ${currentOrder.houseNumber}`;
        this.props.changeCurrentOrderName(title);
        this.props.navigation.setParams({title, currentOrder});
        this.loadItems();
      } else {
        this.props.navigation.navigate('Home', {title: null});
        Alert.alert(
          'Auftrags-ID nicht gefunden',
          'Ein Auftrag mit dieser ID wurde nicht in der Immo-Inspect-Fotos App geladen. ' +
          'Versuchen Sie, die Auftragsliste erneut zu synchronisieren.'
        )
      }
    }
  };
 
  loadItems = () => {
    this.props.triggerLoader();
    let items = [];
    StorageService.getItems().then(i => {
      if (i) {
        i.map((item) => {
          if (
            item != null &&
            item.order === this.props.order &&
            item.user === this.props.user
          ) {
            items.push(item);
          }
        });
        this.props.resetCategoryData();
        this.props.addCategoryData(items);
        this.props.addPhotos(items);
        this.props.checkTotal(items.length);
        this.props.triggerLoader();
        this.getOrder();
        this.checkCategoryValidate();
        this.setState({isShowCategories: true})
      } else return false;
    })
  };
 
  getOrder = () => {
    return StorageService.getItem(this.props.order).then(i => {
      if (!i) {
        let item = {
          id: this.props.order,
          collapse: [
            { order: 2, value: true },
            { order: 5, value: true },
            { order: 7, value: true }
          ],
          validation: false
        };
        return StorageService.saveItem(item);
      } else {
        i.collapse.map(item => this.handleSwitch(item.order, item.value));
      }
    });
  };
 
  deletePhoto = item => {
    Alert.alert(
      "",
      "Sind Sie sicher, dass Sie dieses Foto löschen möchten?",
      [
        {
          text: "Abbrechen",
          onPress: () => console.log("Cancel Pressed"),
          style: "cancel"
        },
        {
          text: "Löschen",
          onPress: () => {
            if (item) {
              StorageService.deleteItem(item.id);
              this.props.deleteItem(item);
              this.props.deletePhoto(item);
            }
          }
        }
      ],
      { cancelable: false }
    );
  };
 
  handleSwitch = (key, state) => {
    this.props.handleCategorySwitch(key, state);
    switch (key) {
      case 2:
        this.props.validateCategory(this.props.data[3].name);
        this.props.validateCategory(this.props.data[4].name);
        break;
      case 5:
        this.props.validateCategory(this.props.data[6].name);
        break;
      case 7:
        this.props.validateCategory(this.props.data[8].name);
        break;
      default:
        break;
    }
  };
 
  deAssignCategory = item => {
    this.props.deAssignCategory(item);
    this.props.deAssignPhotoCategory(item);
    this.props.validateCategory(item.category);
    item.category = "";
    StorageService.saveItem(item);
  };
 
  handleAssignCategory = (item, name) => {
    let oldCategory = item.category;
    item.category = name;
    StorageService.saveItem(item);
    this.props.assignCategory(item, name, oldCategory);
  };
 
  handleAssignDescription = (item, name) => {
    item.description = name;
    StorageService.saveItem(item);
    this.props.assignDescription(item, name);
  };
 
  checkCategoryValidate = () => {
    this.props.dataCategory.forEach((category, key) => {
      if (!([2, 5, 7].indexOf(key) > -1)) {
        this.props.validateCategory(category.name);
      }
    });
  };
 
  saveSwitchToBase = (order, value) => {
    return StorageService.getItem(this.props.order).then(i => {
      i.collapse.map((item,key) => {
        if (order === item.order) {
          i.collapse[key].value = value;
          return StorageService.saveItem(i)
        }
      })
    })
  };
 
  _renderCategoryList = () => {
    const tablet = this.props.tablet;
    return this.props.data
      .map((category, key) => {
        if (category.hasOwnProperty('state')) {
          return (
            <View key={key} style={ styles.containerSwitch}>
              <View style={styles.header}>
                <Text style={styles.textSwitch}>{category.name}</Text>
                <View style={!tablet ? styles.switch : styles.tabletSwitch}>
                  <Switch
                    circleSize={20}
                    value={category.state}
                    onValueChange={() => {
                      this.saveSwitchToBase(key, !category.state)
                        .then(() => {
                          this.handleSwitch(key, !category.state);
                          this.props.validateCategory(category.name);
                        });
                    }}/>
                </View>
              </View>
            </View>
          )
        }
        else {
          return (
           category.visible ?
            <CategoryList
              visible={category.visible}
              handleAssignDescription={this.handleAssignDescription}
              handleAssignCategory={this.handleAssignCategory}
              deAssignCategory={this.deAssignCategory}
              deletePhoto={this.deletePhoto}
              key={key}
              navigate={this.props.navigation}
              category={category.name}
              validate={category.validate}
              photosObj={category.photos}
            />
            : null
        )
      }
    });
  };
 
  render() {
    return (
      <View style={styles.container}>
        <ScrollView>
          {this.props.modelName && this.state.isShowCategories ? this._renderCategoryList() : null}
        </ScrollView>
        <ModalMap orders={[this.props.orderObj]} order={this.props.orderObj} />
      </View>
    );
  }
}
 
const styles = StyleSheet.create({
  container: {
    flex: 1
  },
  header: {
   height:50,
   flex:1,
   flexDirection: 'row',
   justifyContent:'space-between',
   alignItems: 'center',
   zIndex:50,
  },
  textSwitch: {
    minWidth: 75,
    paddingHorizontal: 10,
    fontSize: 16,
    flex: 1
  },
  switch: {
    flex:1,
    marginLeft: 80,
    justifyContent:'center',
    alignItems: 'center',
    width:80
  },
  tabletSwitch: {
    alignItems: 'flex-start',
    width:120,
  },
  containerSwitch: {
    borderBottomColor: 'gray',
    borderBottomWidth: 1,
    zIndex:20,
    flex: 1,
  },
});
 
const mapStateToProps = state => {
  return {
    orderObj: state.App.currentOrderObj,
    data: state.App.dataCategory,
    tablet: state.App.isTablet,
    modelName: state.App.modelName,
    photo: state.App.photo,
    order: state.App.currentOrder,
    orderName: state.App.currentOrderName,
    user: state.Api.userName,
    dataCategory: state.App.dataCategory,
    orders: state.Api.orders,
    isOrderValid: state.App.isOrderValid,
    isReloadingCategories: state.App.isReloadingCategories,
    appStateWasChanged: state.App.appStateWasChanged,
    savedOrders: state.App.savedOrders,
  };
};
 
const mapDispatchToProps = dispatch => {
  return {
    validateCategory: (categoryName) => dispatch(validateCategory(categoryName)),
    changeCurrentOrder: items => dispatch(changeCurrentOrder(items)),
    changeCurrentOrderName: items => dispatch(changeCurrentOrderName(items)),
    deAssignPhotoCategory: item => dispatch(deAssignPhotoCategory(item)),
    resetCategoryData: () => dispatch(resetCategoryData()),
    addPhotos: items => dispatch(addPhotos(items)),
    addCategoryData: items => dispatch(addCategoryData(items)),
    triggerLoader: () => dispatch(triggerLoader()),
    checkTotal: total => dispatch(checkTotal(total)),
    deAssignCategory: item => dispatch(deAssignCategory(item)),
    assignCategory: (item, name, oldCategory) => dispatch(assignCategory(item, name, oldCategory)),
    assignDescription: (item, descriptionName) => dispatch(assignDescription(item, descriptionName)),
    deleteItem: id => dispatch(deleteItem(id)),
    deletePhoto: id => dispatch(deletePhoto(id)),
    handleCategorySwitch: (key, state) => dispatch(handleCategorySwitch(key, state)),
    setReloadingCategoriesFlag: (flag) => dispatch(setReloadingCategoriesFlag(flag))
  }
};
 
export default connect(mapStateToProps, mapDispatchToProps)(ExecutionCategoryView);
