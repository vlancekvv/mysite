import React, { Component } from "react";
import {View, StyleSheet, ScrollView, Text, Alert, BackHandler} from "react-native";
import InspectionExecutionPhotoRow from "./InspectionExecutionPhotoRow";
import { connect } from "react-redux";
import {
  assignCategory, assignDescription, checkTotal, deAssignCategory,
  deAssignPhotoCategory, deleteItem, deletePhoto, toggleAssignPhotoModal, validateCategory
} from "../../actions";
import Slot from './Slot'
import SubCategoryList from './SubCategoryList'
import StorageService from "../../services/StorageService";
import PhotoConfiguration from "./PhotoConfiguration";
 
class InspectionExecutionScreen extends Component {
 
  constructor(props) {
    super(props);
    this.state = {
      modal: false
    }
  }
 
  componentDidMount = () => {
    this.setState({portrait: this.props.orientation === 'PORTRAIT'});
    BackHandler.addEventListener('hardwareBackPress', this.handleBackPress);
  };
 
  componentWillUnmount() {
    BackHandler.removeEventListener('hardwareBackPress', this.handleBackPress);
  }
 
  handleBackPress = () => {
    this.props.navigation.navigate('Home');
    return true;
  };
 
  toggleModal = () => {
    this.setState({ modal: !this.state.modal })
  };
 
  deletePhoto = item => {
    Alert.alert(
      "",
      "Sind Sie sicher, dass Sie dieses Foto löschen möchten?",
      [
        {
          text: "Abbrechen",
          onPress: () => console.log("Cancel Pressed"),
          style: "cancel"
        },
        {
          text: "Löschen",
          onPress: () => {
            if (item) {
              StorageService.deleteItem(item.id)
                .then(() => {
                  this.setState({modal: false});
                  this.props.deleteItem(item);
                  this.props.deletePhoto(item);
                });
            }
          }
        }
      ],
      { cancelable: false }
    );
  };
 
  deAssignCategory = item => {
    this.props.deAssignCategory(item);
    this.props.deAssignPhotoCategory(item);
    this.props.validateCategory(item.category);
    item.category = "";
    StorageService.saveItem(item);
  };
 
  assignPhotoToCategory = (description, photo, categoryKey) => {
    let item = {
      ...photo,
      category_key: categoryKey || this.props.categoryKey,
      description
    };
    let oldCategory = item.category;
    item.category = this.props.neededCategory;
    StorageService.saveItem(item);
    this.props.assignCategory(item, this.props.neededCategory, oldCategory);
    this.props.validateCategory(item.category);
    if (oldCategory.length) {
      this.props.validateCategory(oldCategory);
    }
    if (categoryKey) {
      this.props.toggleAssignPhotoModal();
    }
  };
 
  handleAssignDescription = (item, name) => {
    item.description = name;
    StorageService.saveItem(item);
    this.props.assignDescription(item, name);
  };
 
  _renderSubSlot = (isReorder) => {
    let slot = [], { minPhotos } = this.props.attributes.find(attr => attr.name === this.props.neededCategory).properties;
    for (let index = 0; index < minPhotos; index++) {
      slot.push(<Slot disable = {false} key={10 + index} index={index + 1} assignPhotoToCategory={this.assignPhotoToCategory}/>);
    }
    this.props.dataPhotos.forEach((photoObject, key) => {
      if (photoObject.category === this.props.neededCategory && photoObject.category_key !== 'opt') {
       slot.splice(photoObject.category_key - 1, 1,
        <SubCategoryList
          disable={!isReorder}
          isReorder={isReorder}
          renderSlot={this._renderSubSlot}
          handleAssignDescription={this.handleAssignDescription}
          assignPhotoToCategory={this.assignPhotoToCategory}
          deAssignCategory={this.deAssignCategory}
          key={key}
          navigate={this.props.navigation}
          index={photoObject.category_key}
          item={photoObject}/>)
      }
    });
    return slot;
  };
 
  _renderPhotoRow = () => {
    return this.props.dataPhotos
      .sort((prevPhoto, nextPhoto) => prevPhoto.dateInMilliseconds - nextPhoto.dateInMilliseconds)
      .map((photoObj, key) =>
          <InspectionExecutionPhotoRow
            renderSlot ={this._renderSubSlot}
            handleAssignDescription={this.handleAssignDescription}
            assignPhotoToCategory={this.assignPhotoToCategory}
            deAssignCategory={this.deAssignCategory}
            toggleModal={this.toggleModal}
            key={key}
            navigate={this.props.navigation}
            item={photoObj}/>
    );
  };
 
  render() {
    const tablet = this.props.tablet;
    return (
      <View style={!tablet ? styles.container : styles.containerTablet}>
        <View style={styles.list}>
        { (this.props.dataPhotos.length === 0) ?
          <View style={styles.centerText}>
            <Text>Noch keine Fotos vorhanden</Text>
          </View>
          :
          <ScrollView
            contentContainerStyle={styles.listView}
            ref="list"
            enableEmptySections={true}>
            {this._renderPhotoRow()}
          </ScrollView>
           }
        </View>
        <PhotoConfiguration
          renderSlot={this._renderSubSlot}
          showModal={this.state.modal}
          handleAssignDescription={this.handleAssignDescription}
          assignPhotoToCategory={this.assignPhotoToCategory}
          deAssignCategory={this.deAssignCategory}
          deletePhoto={this.deletePhoto}
          toggleModal={this.toggleModal}
          navigate={this.props.navigation}/>
      </View>
    );
  }
}
 
const styles = StyleSheet.create({
  container: {
    flex: 1,
    flexDirection: "column",
    justifyContent: "center"
  },
  containerTablet: {
    flex: 1,
    flexDirection: "column",
    justifyContent: "center"
  },
  list: {
    flex: 1
  },
  listView: {
    flexDirection: "row",
    flexWrap: "wrap"
  },
  centerText: {
    flex: 1,
    alignItems: 'center',
    justifyContent: 'center',
  }
});
 
const mapStateToProps = state => {
  return {
    dataPhotos: state.App.dataPhotos,
    tablet: state.App.isTablet,
    neededCategory: state.App.neededCategory,
    orders: state.Api.orders,
    order: state.App.currentOrder,
    modelName: state.App.modelName,
    attributes: state.App.attributes,
    categoryKey: state.App.categoryKey,
    isShowAssignPhotoModal: state.App.isShowAssignPhotoModal
  };
};
 
const mapDispatchToProps = dispatch => {
  return {
    checkTotal: total => dispatch(checkTotal(total)),
    deAssignPhotoCategory: item => dispatch(deAssignPhotoCategory(item)),
    deAssignCategory: item => dispatch(deAssignCategory(item)),
    assignCategory: (item, name, oldCategory) => dispatch(assignCategory(item, name, oldCategory)),
    assignDescription: (item, descriptionName) => dispatch(assignDescription(item, descriptionName)),
    validateCategory: (categoryName, value) => dispatch(validateCategory(categoryName, value)),
    deleteItem: id => dispatch(deleteItem(id)),
    deletePhoto: id => dispatch(deletePhoto(id)),
    toggleAssignPhotoModal: () => dispatch(toggleAssignPhotoModal())
  };
};
 
export default connect(mapStateToProps, mapDispatchToProps)(InspectionExecutionScreen);
