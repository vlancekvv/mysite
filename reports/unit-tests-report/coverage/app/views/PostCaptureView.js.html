import React, {Component} from 'react'
import {
  StyleSheet,
  View,
  Dimensions,
  Alert,
  BackHandler
} from 'react-native'
import { connect } from 'react-redux'
import _ from 'lodash'
 
import StorageService from '../services/StorageService'
import ImageListView from './ImageListView'
import { PostCapture } from '../../packages/photo-module'
import errorHandler from '../utils/ErrorHandler'
import {triggerLoader, addPhoto, editPhoto} from '../actions';
import ImageService from '../../packages/photo-module/services/ImageService';
 
class PostCaptureView extends Component {
 
  constructor(props) {
    super(props);
    this.state = {
      portrait: false,
      rotate90deg: 0
    }
  }
 
  componentDidMount = () => {
    this.setState({portrait: this.props.orientation === 'PORTRAIT'});
    BackHandler.addEventListener('hardwareBackPress', this.handleBackPress);
  };
 
  componentWillUnmount() {
    BackHandler.removeEventListener('hardwareBackPress', this.handleBackPress);
  }
 
  handleBackPress = () => {
    this.props.navigation.navigate('CategoryView', {title: this.props.orderName, id: this.props.currentOrder});
    return true;
  };
 
  saveBlurredImage = async (sizes = {}, screen) => {
    const { payload, oldPayload, isAddedPhoto} = this.props.navigation.state.params;
    payload.img = { thumbnail: sizes.thumbnail};
    let originalImageItem;
    const storageItem = await StorageService.saveItem(payload);
    const unblurredRotatedOriginal = await ImageService.resizeOriginal(oldPayload.img.original, this.props.rotate90deg);
    delete sizes.thumbnail;
    if (isAddedPhoto) {
      originalImageItem = await StorageService.updateOriginal(storageItem.id, sizes.original, unblurredRotatedOriginal);
      this.props.editPhoto(payload)
    } else {
      originalImageItem = await StorageService.saveOriginal(storageItem.id, sizes.original, unblurredRotatedOriginal);
      this.props.addPhoto(payload);
    }
    this.navigateToScreen(screen, storageItem, originalImageItem);
  };
 
  saveUnblurredImage = async (sizes = {}, screen) => {
    const { payload, isAddedPhoto } = this.props.navigation.state.params;
    const newPayload = {...payload, img: { thumbnail: sizes.thumbnail }};
    let originalImageItem;
    const storageItem = await StorageService.saveItem(newPayload);
    delete sizes.thumbnail;
    if (isAddedPhoto) {
      originalImageItem = await StorageService.updateOriginal(storageItem.id, sizes.original);
      this.props.editPhoto(newPayload)
    } else {
      originalImageItem = await StorageService.saveOriginal(storageItem.id, sizes.original);
      this.props.addPhoto(newPayload);
    }
    this.navigateToScreen(screen, storageItem, originalImageItem);
  };
 
  navigateToScreen = (screen, storageItem, originalImageItem) => {
    this.props.triggerLoader();
    if (storageItem && originalImageItem) this.props.navigation.navigate(screen, {
      title: this.props.orderName,
      id: this.props.currentOrder
    });
    else errorHandler(new Error("Cannot save data to storage!"));
  };
 
  editImage = async (rotate90deg) => {
    let unblurredOriginal = false;
    const { databaseId } = this.props.navigation.state.params;
    if (databaseId) {
      unblurredOriginal = await StorageService.getUnblurredOriginal(databaseId);
      if (unblurredOriginal) {
        Alert.alert(
          "Zensur aufheben",
          "Sollen alle zensierten Bereiche wieder kenntlich gemacht werden? Indem Sie fortfahren, werden alle Zensuren aufgehoben – Sie können die Bereiche erneut unkenntlich machen. ",
          [
            {text: 'Fortfahren', onPress: () => {
              const payload = _.cloneDeep(this.props.navigation.state.params.payload);
              payload.img.original = unblurredOriginal;
              this.navigateToBlurView(payload, rotate90deg, true);
            } },
            {text: 'Abbrechen', onPress: () => {}, style: 'cancel'},
          ],
        )
      } else this.firstEdit(rotate90deg)
    } else this.firstEdit(rotate90deg)
  };
 
  firstEdit = (rotate90deg) => {
    const { payload, isAddedPhoto } = this.props.navigation.state.params;
    this.navigateToBlurView(payload, rotate90deg, isAddedPhoto);
  };
 
  navigateToBlurView = (payload, rotate90deg, isAddedPhoto) => {
    this.props.navigation.navigate('BlurView', {
      payload,
      rotate90deg,
      title: this.props.orderName,
      id: this.props.currentOrder,
      isAddedPhoto
    })
  };
 
  navigate = (where) => {
    this.props.navigation.navigate(where, {title: this.props.orderName, id: this.props.currentOrder})
  };
 
  render() {
    const { payload, oldPayload } = this.props.navigation.state.params;
    const TABLET = this.props.tablet;
    const PORTRAIT = this.props.orientation === "PORTRAIT";
    const tabletTale = (TABLET && !PORTRAIT) ? 'Tablet' : '';
    const tabletPortraitTale = (TABLET && PORTRAIT) ? 'Portrait' : '';
    const imageListView = (
      <View style={styles[`imageContainer${tabletPortraitTale}`]}>
        <ImageListView navigation = {this.props.navigation}/>
      </View>
    );
    return (
      <View style={ styles[`container${tabletTale}`]}>
        { TABLET && !PORTRAIT && imageListView }
        <PostCapture
          tablet = { this.props.tablet }
          orientation = { this.props.orientation }
          original = { payload.img.original }
          triggerLoader = { this.props.triggerLoader }
          save = { oldPayload ? this.saveBlurredImage  : this.saveUnblurredImage }
          onEditImage = { this.editImage }
          navigate = { this.navigate }
          onError = {(err) => {
            this.props.triggerLoader();
            errorHandler(err)
          }}
        />
        { TABLET && PORTRAIT && imageListView }
      </View>
    );
  }
}
 
const { width } = Dimensions.get('window');
 
 
const styles = StyleSheet.create({
  container: {
    flex: 1,
    flexDirection:  'column',
    justifyContent: 'space-between',
    backgroundColor: '#fff'
  },
  containerTablet: {
    flex: 1,
    backgroundColor: '#fff',
    flexDirection:  'row',
  },
  imageContainerPortrait: {
    position:'absolute',
    bottom: 0,
    width: width,
  },
  imageContainer: {
    width: (Dimensions.get('window').width / 6)
  }
});
 
const mapStateToProps = state => {
  return {
    tablet: state.App.isTablet,
    orientation: state.App.orientation,
    orderName: state.App.currentOrderName,
    currentOrder: state.App.currentOrder
  }
};
 
const mapDispatchToProps = dispatch => {
  return{
    triggerLoader: () => dispatch(triggerLoader()),
    addPhoto: photo => dispatch(addPhoto(photo)),
    editPhoto: photo => dispatch(editPhoto(photo))
  }
};
 
export default connect(mapStateToProps, mapDispatchToProps)(PostCaptureView)
