import React, { Component } from 'react';
import { View, StyleSheet, TouchableOpacity, Text, Alert } from 'react-native';
import { connect } from 'react-redux';
import Icon from 'react-native-vector-icons/MaterialCommunityIcons';
 
import { changeCurrentOrder, changeCurrentOrderName } from '../../actions';
import { removeOrder } from '../../actions/backend';
 
class RenderOrder extends Component {
 
  constructor(props) {
    super(props);
    this.state = {
      lastPressTime: 0
    }
  }
 
  handleActiveOrder = (order) => {
    this.props.changeCurrentOrder(order);
    this.props.navigation.setParams({title: `${order.street} ${order.houseNumber}`, id: order.goVID, isEditButtonDisplayed: true});
    const currentPressTime = Date.now();
    return currentPressTime - this.state.lastPressTime < 200 ?
      this.navigateToCategory(order) :
      this.selectOrder(this.props.item, currentPressTime);
  };
 
  selectOrder = (order, currentPressTime) => {
    this.setState({lastPressTime: currentPressTime});
    const title = order.street + ' ' + order.houseNumber;
    this.props.changeCurrentOrderName(title);
  };
 
  navigateToCategory = (order) => {
    const title = `${order.street} ${order.houseNumber}`;
    this.props.changeCurrentOrder(order);
    this.props.changeCurrentOrderName(title);
    this.props.navigation.navigate('CategoryView', {title, id: order.goVID})
  };
 
  deleteOrder = (order) => {
    if (!this.props.useLiveServer || order.status === 50) {
      Alert.alert(
        `Möchten Sie den Auftrag ${order.street} ${order.houseNumber} permanent löschen?`,
        'Bitte beachten Sie, dass nach dem Löschen eines Auftrags im Archiv die Wiederbearbeitung mit bereits ' +
        'aufgenommenen Fotos im Reklamationsfall nicht möglich ist. Die App löscht automatisch die Aufträge, die sicher ' +
        'gelöscht werden können.',
        [
          {text: 'Ja', onPress: () => this.props.removeOrder(order.goVID), style: 'destructive'},
          {
            text: 'Nein', onPress: () => {
            }, style: 'cancel'
          }
        ],
        {cancelable: false})
    }
  };
 
  getDaysToDeletion = (expiration) => {
    return `Tage bis zum Löschen: ${Math.round((expiration - Date.now()/1000)/3600/24)}`;
  };
 
  render() {
    let order = this.props.item;
    const tablet = this.props.tablet;
    return (
      <View>
        <View style={!(this.props.order === order.goVID) ? null : styles.activeBorder}>
          <TouchableOpacity onPress={() => this.props.isArchiveOrder ? this.deleteOrder(order) : this.handleActiveOrder(order)}
                            onLongPress={() => !this.props.isArchiveOrder && this.navigateToCategory(order)}
                            style={!tablet  ? styles.container : styles.containerTablet}
                            accessibilityLabel={order.statusLabel}>
            <View style={styles.textContainer}>
              <Text numberOfLines={1} style={styles.leftTextBold}>{order.goVID} {order.inspectionModelLabel}</Text>
              <Text numberOfLines={1} style={styles.rightTextBold}>{order.postalCode} {order.city}</Text>
            </View>
            <View style={{...styles.textContainer, justifyContent:  order.status < 37 || order.status === 50 ? 'space-between' : 'flex-end'}}>
              {
                order.status < 37 || order.status === 50 ?
                  <Text numberOfLines={1} style={styles.leftText}>
                    {this.props.isArchiveOrder ? this.getDaysToDeletion(order.expiration)  : order.dueLabel}
                  </Text>
                  : null
              }
              <Text numberOfLines={1} style={styles.rightText}>{order.street} {order.houseNumber}</Text>
            </View>
            <View style={styles.textContainer}>
              <View style={styles.statusContainer}>
                <Text numberOfLines={1} style={styles.leftText}>
                  {order.statusLabel}
                </Text>
                {order.photos? <Icon size={15} name={'camera'} style={{marginLeft: 5}}/> : null}
              </View>
 
              <Text numberOfLines={1} style={styles.rightText}>{order.buildingType}</Text>
            </View>
          </TouchableOpacity>
        </View>
      </View>
    )
  }
}

    alignItems: 'center'
  }

const mapStateToProps = state => {
  return {
    tablet: state.App.isTablet,
    order: state.App.currentOrder,
    useLiveServer: state.App.useLiveServer
  }
};
 
const mapDispatchToProps = dispatch => {
  return {
    changeCurrentOrder: (order) => dispatch(changeCurrentOrder(order)),
    changeCurrentOrderName: (orderName) => dispatch(changeCurrentOrderName(orderName)),
    removeOrder: id => dispatch(removeOrder(id))
  }
};
 
export default connect(mapStateToProps, mapDispatchToProps)(RenderOrder)
