import React, { Component } from 'react'
import {View, StyleSheet, TouchableOpacity, Text, Alert} from 'react-native'
import { Header } from 'react-navigation';
import Icon from 'react-native-vector-icons/MaterialCommunityIcons'
import Modal from 'react-native-modal';
import { connect } from 'react-redux';
 
import {handleLogout} from '../../actions/backend';
import {toggleMenu} from '../../actions';
import StorageService from '../../services/StorageService';
 
export class Menu extends Component {
 
  logout = () => {
    Alert.alert(
      'Möchten Sie sich abmelden?',
      `- Die eingegebenen Daten bleiben in der App zwischengespeichert.` +
      `\n\n- Bei einer erneuten Anmeldung ist eine Internetverbindung erforderlich.`,
      [...this.getButtons()],
      { cancelable: true }
    )
  };
 
  getButtons = () => {
     return !this.props.useLiveServer ?
       [
         {
           text: 'Abmelden',
           onPress: () => StorageService.updateAuthorizationFalse('mail', 'password')
             .then(() => this.props.handleLogout())
         },
         { text: 'Abmelden und Daten löschen', onPress: () => this.hardLogout(),style: 'destructive' },
         { text: 'Abbrechen', onPress: () => console.log('Cancel Pressed'), style: 'cancel' }
       ] :
       [
         {
           text: 'Abmelden',
           onPress: () => StorageService.updateAuthorizationFalse('mail', 'password')
             .then(() => this.props.handleLogout())
         },
         { text: 'Abbrechen', onPress: () => console.log('Cancel Pressed'), style: 'cancel' }
       ]
  };
 
  hardLogout = () => {
    Alert.alert(
      'Sind Sie sicher, dass Sie auch alle Daten und Fotos löschen möchten?',
      null,
      [
        { text: 'Nein', onPress: () => {}, style: 'cancel' },
        {
          text: 'Ja',
          onPress: () => StorageService.updateAuthorizationFalse('mail', 'password')
            .then(() => StorageService.clear())
            .then(() => this.props.handleLogout()),
          style: 'destructive'
        },
      ],
      { cancelable: false }
    )
  };
 
  render() {
    return (
      <Modal animationInTiming={1}
             animationOutTiming={1}
             onBackdropPress={this.props.toggleMenu}
             backdropOpacity={0}
             isVisible={this.props.isShowMenu}
             style={{...styles.menu}}>
        <View style={styles.triangle}/>
        <View style={styles.modalContent}>
          <TouchableOpacity style={styles.modalClick}
                            accessibilityLabel="openArchiveButton"
                            onPress={() => {
                              this.props.navigation.navigate('ArchiveOrders');
                              this.props.toggleMenu();
                            }}>
            <Icon name='archive'
                  size={25}
                  style={styles.menuIcon}
                  color={'#3c5296'}
                  backgroundColor='transparent'/>
            <Text style={styles.modalText}>Archiv</Text>
          </TouchableOpacity>
          <TouchableOpacity style={[styles.modalClick, styles.lastModalClick]}
                            onPress={() => {
                              setTimeout(() => this.logout(), 40);
                              this.props.toggleMenu();
                            }}>
            <Icon name='logout-variant'
                  size={25}
                  style={styles.menuIcon}
                  color={'#3c5296'}
                  backgroundColor='transparent'/>
            <Text style={styles.modalText}>Abmelden</Text>
          </TouchableOpacity>
        </View>
      </Modal>
    )
  }
}
 
const styles = StyleSheet.create({
  modalClick: {
    width: '100%',
    height: 50,
    display: 'flex',
    flexDirection: 'row',
    justifyContent: 'center',
    alignItems: 'center',
    marginTop: 0.5,
    zIndex: 10,
    borderBottomColor: '#bababa',
    borderBottomWidth: 0.5
  },
  lastModalClick: {
    borderBottomWidth: 0
  },
  modalText: {
    fontSize: 18,
    color: '#3c5296'
  },
  modalContent: {
    height: 'auto',
    width: 300,
    backgroundColor: '#eaeaea',
    justifyContent: 'center',
    alignItems: 'center',
    borderRadius: 10,
    borderColor: '#bababa'
  },
  menu: {
    top: Header.HEIGHT + 15,
    margin: 0,
    left: 5,
    alignItems: 'flex-start',
    justifyContent: 'flex-start',
    position: 'absolute'
  },
  menuIcon: {
    marginRight: 10
  },
  triangle: {
    position: 'absolute',
    top: -10,
    left: 5,
    width: 0,
    height: 0,
    backgroundColor: 'transparent',
    borderStyle: 'solid',
    borderLeftWidth: 20,
    borderRightWidth: 20,
    borderBottomWidth: 20,
    borderLeftColor: 'transparent',
    borderRightColor: 'transparent',
    borderBottomColor: '#eaeaea'
  }
});
 
const mapStateToProps = state => {
  return {
    isShowMenu: state.App.isShowMenu,
    useLiveServer: state.App.useLiveServer
  }
};
 
const mapDispatchToProps = dispatch => {
  return {
    toggleMenu: () => dispatch(toggleMenu()),
    handleLogout: () => dispatch(handleLogout())
  }
};
 
export default connect(mapStateToProps, mapDispatchToProps)(Menu)
