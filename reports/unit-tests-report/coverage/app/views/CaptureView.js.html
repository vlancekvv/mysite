import React, { Component } from "react"
import { StackActions, NavigationActions } from 'react-navigation';
import { connect } from "react-redux"
import { Capture } from "../../packages/photo-module"
import { triggerLoader,changeCurrentOrder,changeCurrentOrderName } from "../actions/index"
import errorHandler from "../utils/ErrorHandler"
import {setReloadingCategoriesFlag} from "../actions";
import {Alert, BackHandler} from "react-native";
 
class CaptureView extends Component {
 
  constructor(props) {
    super(props);
    this.state = {
      isAbortHandleUrl: false
    }
  }
 
  componentDidMount = () => {
    this.props.setReloadingCategoriesFlag(false);
    this.handleUrl();
    BackHandler.addEventListener('hardwareBackPress', this.handleBackPress);
  };
 
  componentDidUpdate() {
    if (this.props.appStateWasChanged && this.props.orders.length && !this.state.isAbortHandleUrl) {
      this.props.setReloadingCategoriesFlag(false);
      this.handleUrl();
    }
  }
 
  componentWillUnmount() {
    BackHandler.removeEventListener('hardwareBackPress', this.handleBackPress);
  }
 
  handleBackPress = () => {
    this.props.navigation.navigate('CategoryView', {title: this.props.orderName, id: this.props.currentOrder});
    return true;
  };
 
  handleUrl = () =>  {
    let urlId = +this.props.navigation.state.params.id;
    if (urlId) {
      const currentOrder = this.props.orders.find(order => order.goVID === urlId);
      if (currentOrder) {
        this.props.changeCurrentOrder(currentOrder);
        const title = `${currentOrder.street} ${currentOrder.houseNumber}`;
        this.props.changeCurrentOrderName(title);
        this.props.navigation.setParams({title, currentOrder});
      } else {
        this.setState({isAbortHandleUrl: true});
        this.props.navigation.navigate('Home', {title: null});
        Alert.alert(
          'Auftrags-ID nicht gefunden',
          'Ein Auftrag mit dieser ID wurde nicht in der Immo-Inspect-Fotos App geladen. ' +
          'Versuchen Sie, die Auftragsliste erneut zu synchronisieren.'
        )
      }
    }
  };
 
  render() {
    return (
      <Capture
        tablet={ this.props.tablet }
        triggerLoader={ this.props.triggerLoader }
        orientation = { this.props.orientation }
        onPictureCaptured={ this.onPictureCaptured }
        navigation={this.props.navigation}
        onError={ () => errorHandler()}
      />
    );
  }
 
  onPictureCaptured = ( payload ) => {
    const resetAction = StackActions.reset({
      index: 1,
      actions: [
        NavigationActions.navigate({ routeName: 'CategoryView', params:
            {
              title: this.props.orderName,
              id: this.props.currentOrder
            }
        }),
        NavigationActions.navigate({ routeName: 'PostCaptureView', params:
            {
              payload,
              title: this.props.orderName,
              id: this.props.currentOrder
            }
        }),
      ],
    });
    this.props.navigation.dispatch(resetAction);
  }
 
}
 
const mapStateToProps = state => {
  return {
    orientation: state.App.orientation,
    tablet: state.App.isTablet,
    orderName: state.App.currentOrderName,
    currentOrder: state.App.currentOrder,
    orders: state.Api.orders,
    appStateWasChanged: state.App.appStateWasChanged
  };
};
 
const mapDispatchToProps = dispatch => {
  return {
    triggerLoader: () => dispatch(triggerLoader()),
    changeCurrentOrder: items => dispatch(changeCurrentOrder(items)),
    changeCurrentOrderName: items => dispatch(changeCurrentOrderName(items)),
    setReloadingCategoriesFlag: (flag) => dispatch(setReloadingCategoriesFlag(flag))
  };
};
 
export default connect(mapStateToProps, mapDispatchToProps)(CaptureView);
