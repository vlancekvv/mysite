import React, { Component } from 'react';
import { View, TouchableOpacity, StyleSheet, Text, Linking, Alert, Platform } from 'react-native';
import MapView, { Marker } from 'react-native-maps';
 
/**
 * Initial coordinates for map
 * @type {{latitude: number, longitude: number, latitudeDelta: number, longitudeDelta: number}}
 */
export const initialCoords = {
  latitude: 51.1657,
  longitude: 10.4515,
  latitudeDelta: 10,
  longitudeDelta: 10
};
 
/**
 * Map component
 */
export class MapComponent extends Component {
 
  /**
   * Schedules a external app should be opened
   */
  openExternalMap() {
    let { latitude, longitude } = this.props.selectedOrder, url;
    if (latitude) {
      url = `https://www.google.com/maps/search/?api=1&query=${latitude},${longitude}`;
    } else {
      url = `https://www.google.com/maps/@${initialCoords.latitude},${initialCoords.longitude},6z`
    }
    return Linking.canOpenURL(url).then(supported => {
      if (!supported) {
        return Alert.alert('Cannot open external app with this coordinates')
      } else {
        return Linking.openURL(url);
      }
    }).catch(() => Alert.alert('Cannot open external app with this coordinates'));
  }
 
  /**
   *  Handle map ready event
   */
  handleMapReady() {
    this.setState({isMapReady: true});
  }
 
  /**
   * Should render map
   */
  render() {
    const { selectedOrder, orders, isModal } = this.props;
    return (
      <View style={{...styles.container, borderRadius: isModal ? 10 : 0}}>
        <MapView
          style={{...styles.container, borderRadius: isModal ? 10 : 0}}
          loadingEnabled
          initialRegion={initialCoords}
          mapType={Platform.OS === 'ios' ? 'hybrid' : 'standard'}
          onMapReady={this.handleMapReady.bind(this)}
          region={ selectedOrder.latitude ? {
            latitude: selectedOrder.latitude,
            longitude: selectedOrder.longitude,
            latitudeDelta: 0.002,
            longitudeDelta: 0.002
          } : initialCoords}>
          {
            orders.map((order, key) =>
              <Marker
                key={key}
                coordinate={{
                  latitude: order.latitude,
                  longitude: order.longitude,
                }}
                title={order.title}
              />
            )
          }
        </MapView>
        {this.state && this.state.isMapReady && (
          <TouchableOpacity style={styles.button} onPress={this.openExternalMap.bind(this)}>
            <Text style={{fontSize: 16}}>In externer Karte Ã¶ffnen</Text>
          </TouchableOpacity>
        )}
      </View>
    )
  }
}
 
const styles = StyleSheet.create({
  container: {
    flex: 1,
    width: '100%'
  },
  button: {
    display: 'flex',
    alignItems: 'center',
    justifyContent: 'center',
    backgroundColor: '#fff',
    width: 'auto',
    height: 'auto',
    padding: 5,
    position: 'absolute',
    bottom: 20,
    right: 20,
    borderRadius: 5
  }
});
