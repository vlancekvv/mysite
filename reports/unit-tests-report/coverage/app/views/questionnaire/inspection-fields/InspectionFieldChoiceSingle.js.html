import React from 'react';
import PropTypes from 'prop-types';
import { StyleSheet, Text, TouchableOpacity } from 'react-native';
 
import { getInspectionField } from '../InspectionField';
import { modalStyle } from '../../../shared/styles';
import { EditFieldModal } from './EditFieldModal';
 
/**
 * Inspection field single choice component
 */
export function InspectionFieldChoiceSingle({value, renderDefaultValue, editingHandler, startHandler, choices, ...props}) {
 
  const choicesList = choices.map((name, key) =>
    <TouchableOpacity key={key}
                      style={modalStyle.modalClick}
                      onPress={() => editingHandler(value !== name ? name : '', props.confirmHandler)}>
      <Text style={{...modalStyle.modalText, color: value === name ? '#007AFF' : 'black'}}>{name}</Text>
    </TouchableOpacity>
  );
 
  return (
    <TouchableOpacity style={styles.choiceContainer} onPress={startHandler}>
      {renderDefaultValue()}
      {
        value ?
          <Text style={styles.text}>{value}</Text> :
          <Text style={{...styles.text, color: '#c7c7c7'}}>Einfachauswahl</Text>
      }
      {props.isEditing && <EditFieldModal {...props} renderContent={choicesList} />}
    </TouchableOpacity>
  );
}
 
InspectionFieldChoiceSingle.propTypes = {
  value: PropTypes.string,
  editingHandler: PropTypes.func,
  confirmHandler: PropTypes.func,
  startHandler: PropTypes.func,
  renderDefaultValue: PropTypes.func,
  choices: PropTypes.arrayOf(PropTypes.string),
  isEditing: PropTypes.bool
};
 
const styles = StyleSheet.create({
  choiceContainer: {
    flex: 1,
    width: '100%',
    justifyContent: 'center',
    alignItems: 'center'
  },
  text: {
    fontSize: 18
  }
});
 
export default getInspectionField(InspectionFieldChoiceSingle);
