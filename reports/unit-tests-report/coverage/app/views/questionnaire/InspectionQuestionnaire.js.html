import React, { Component } from 'react';
import { ScrollView, StyleSheet, View } from 'react-native';
import { connect } from 'react-redux';
 
import InspectionFieldBoolean from './inspection-fields/InspectionFieldBoolean';
import InspectionFieldText from './inspection-fields/InspectionFieldText';
import InspectionFieldDate from './inspection-fields/InspectionFieldDate';
import InspectionFieldChoiceSingle from './inspection-fields/InspectionFieldChoiceSingle';
import InspectionFieldChoiceMultiple from './inspection-fields/InspectionFieldChoiceMultiple';
import HelpButtons from './HelpButtons';
 
/**
 * Inspection questionnaire screen component
 */
export class InspectionQuestionnaire extends Component {
 
  constructor(props) {
    super(props);
    this.renderField = this.renderField.bind(this);
  }
 
  /**
   * Schedules a field component should be returned
   * @returns Special field component by type
   */
  getFieldComponent(field) {
    switch (field.fieldType) {
      case 'Nummer': return <InspectionFieldText isNumber={true} field={field} />;
      case 'Text': return <InspectionFieldText field={field} />;
      case 'Text Multiline': return <InspectionFieldText field={field} />;
      case 'Datum': return <InspectionFieldDate field={field} />;
      case 'Ja/Nein': return <InspectionFieldBoolean field={field} />;
      case 'Einfachauswahl': return <InspectionFieldChoiceSingle field={field} />;
      case 'Mehrfachauswahl': return <InspectionFieldChoiceMultiple field={field} />;
    }
  };
 
  /**
   * Schedules a field component should be rendered
   */
  renderField(field, key) {
    return <View key={key}>{this.getFieldComponent(field)}</View>;
  }
 
  /**
   * Schedules a inspection questionnaire screen should be rendered
   */
  render() {
    return (
      <View style={styles.fieldsContainer}>
        <ScrollView style={styles.fieldsContainer} contentInset={{bottom: 300}}>
          {
            this.props.fields.map(this.renderField)
          }
        </ScrollView>
        <HelpButtons />
      </View>
    )
  }
}
 
const styles = StyleSheet.create({
  fieldsContainer: {
    flex: 1
  }
});
 
const mapStateToProps = state => {
  return {
    fields: state.questionnaire.fields
  };
};
 
export default connect(mapStateToProps)(InspectionQuestionnaire);
