import testData from '../../../shared/testData';
 
export const defaultState = {
  fields: testData.fields,
  nextFieldId: null,
  fieldId: null,
  showHelpButtons: false
};
 
export const questionnaire = (state = defaultState, action) => {
  switch(action.type) {
    case 'UPDATE_ATTRIBUTE':
      return {
        ...state,
        fields: state.fields.map(field => field.id === action.id ? {...field, selectedValue: action.value} : field)
      };
    case 'SELECT_ATTRIBUTE':
      return {
        ...state,
        fieldId: action.field.id
      };
    case 'SELECT_PREV_ATTRIBUTE':
      const currentIndex = state.fields.findIndex(field => field.id === state.fieldId);
      return {
        ...state,
        nextFieldId: state.fields[(!currentIndex ? 1 : currentIndex) - 1].id
      };
    case 'SELECT_NEXT_ATTRIBUTE':
      const index = state.fields.findIndex(field => field.id === state.fieldId);
      return {
        ...state,
        nextFieldId: state.fields[index + 1 === state.fields.length ? index : index + 1].id
      };
    case 'RESET_NEXT_ATTRIBUTE':
      return {
        ...state,
        nextFieldId: null
      };
    case 'TOGGLE_BUTTONS':
      return {
        ...state,
        showHelpButtons: !state.showHelpButtons
      };
    default:
      return state
  }
};
