import _ from 'lodash';
import Fuse from "fuse.js";
 
export const defaultState = {
  authorized: null,
  orders: [],
  searchedOrders: null,
  userName: '',
  passwordError: false,
  progressLoading: null,
  lastSynchronization: null,
  refreshingOrders: false,
  isManualUpdating: false,
  synchronizationError: null
};
 
export const Api = (state = defaultState, action) => {
  switch (action.type) {
    case 'LOGGED_IN_SUCCESSFULLY':
      return { ...state, authorized: true};
    case 'LOGIN_FAILED':
      return { ...state, authorized: false };
    case 'SET_ORDERS':
      return { ...state, orders: action.orders};
    case 'UPDATE_ORDERS':
      return {
        ...state,
        orders: _.uniqBy(action.orders.concat(state.orders), 'goVID'),
        synchronizationError: null
      };
    case 'LOGOUT':
      return { ...state, authorized: false };
    case 'SET_USER':
      return { ...state, userName: action.userName};
    case 'PASSWORD_ERROR':
      return { ...state, passwordError: !action.passwordError};
    case 'SET_PROGRESS':
      return { ...state, progressLoading: action.progress};
    case 'RESET_PROGRESS':
      return { ...state, progressLoading: null};
    case 'TOGGLE_REFRESHING':
      return { ...state,
        refreshingOrders: !state.refreshingOrders,
        isManualUpdating: false
      };
    case 'SAVE_UPDATING_TIME':
      return { ...state, lastSynchronization: action.time};
    case 'SET_MANUAL_UPDATING_FLAG':
      return { ...state, isManualUpdating: true};
    case 'REMOVE_ORDER':
      return {
        ...state,
        orders: state.orders.filter(order => order.goVID !== action.id)
      };
    case "HANDLE_SYNCHRONIZATION_ERROR":
      return {...state, synchronizationError: action.error};
    case "SEARCH_ORDERS":
      const fuse = new Fuse(state.orders, action.options);
      return {
        ...state,
        searchedOrders: action.text.length ? fuse.search(action.text) : null
      };
    default:
      return state
  }
};
