export function login(userName, userPassword) {
  store.dispatch(triggerLoader());
  return dispatch => ApiService.login(userName, userPassword)
    .then(response => {
      if (response.headers['content-type'].match('text/html')) {
        throw new Error('Error: Request failed with status code 401');
      } else if (response.status === 200) {
        dispatch(setUser(userName));
      } else {
        throw new Error(response.statusText);
      }
      return StorageService.putUserEmail(userName)
        .then(() => StorageService.getAuthorization())
        .then(auth => !auth ? StorageService.setAuthorization(true,userName,userPassword) :
          StorageService.updateAuthorizationTrue(userName,userPassword))
        .then(() => dispatch(loginSuccessfully()))
        .then(() => dispatch(getOrders(OrderService.createUniqueUpdatingTime())))
        .then(() => dispatch(triggerLoader()))
        })
    .catch(err => {
      if (err.message === 'Error: Request failed with status code 401' || err.message === 'Error: Request failed with status code 505') {
        dispatch(passwordError());
      }
      dispatch(triggerLoader());
      Alert.alert(err.message);
    })
}
