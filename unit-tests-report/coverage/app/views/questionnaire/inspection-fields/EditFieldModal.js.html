import React from 'react';
import PropTypes from 'prop-types';
import {ScrollView, StyleSheet, Text, TouchableOpacity, View} from 'react-native';
import Modal from 'react-native-modal';
import Icon from 'react-native-vector-icons/MaterialCommunityIcons';
 
import { modalStyle } from '../../../shared/styles';
 
/**
 * Edit field modal component
 */
export function EditFieldModal({fieldName, confirmHandler, isEditing, cancel, renderContent, reset, selectNextField, selectPrevField}) {
  return (
    <Modal isVisible={isEditing} onBackdropPress={confirmHandler} style={modalStyle.modal}>
      <View style={modalStyle.modalContent}>
        <View style={styles.header}>
          <Text style={{paddingVertical: 10, fontSize: 18}}>{fieldName}</Text>
          <Icon
            style={{...styles.headerIcon, right: 50}}
            onPress={selectPrevField}
            size={25}
            color="black"
            name="arrow-up"
          />
          <Icon
            style={{...styles.headerIcon, right: 10}}
            onPress={selectNextField}
            size={25}
            color="black"
            name="arrow-down"
          />
        </View>
        <ScrollView style={{width: '100%'}}>
          {renderContent}
        </ScrollView>
        <TouchableOpacity style={modalStyle.cancelModalClick}
                          onPress={cancel}>
          <Text style={modalStyle.cancelModalText}>Abbrechen</Text>
        </TouchableOpacity>
        <TouchableOpacity style={modalStyle.cancelModalClick}
                          onPress={reset}>
          <Text style={modalStyle.cancelModalText}>Reset</Text>
        </TouchableOpacity>
        <TouchableOpacity style={modalStyle.cancelModalClick}
                          onPress={confirmHandler}>
          <Text style={modalStyle.cancelModalText}>Save</Text>
        </TouchableOpacity>
      </View>
    </Modal>
  )
}
 
EditFieldModal.propTypes = {
  confirmHandler: PropTypes.func,
  fieldName: PropTypes.string,
  cancel: PropTypes.func,
  isEditing: PropTypes.bool,
  reset: PropTypes.func,
  selectNextField: PropTypes.func,
  selectPrevField: PropTypes.func
};
 
const styles = StyleSheet.create({
  header: {
    height: 50,
    width: '100%',
    paddingHorizontal: 10,
    display: 'flex',
    alignItems: 'center',
    justifyContent: 'center',
    borderBottomColor: '#bdbdbd',
    borderBottomWidth: 0.5
  },
  headerIcon: {
    position: 'absolute',
    top: 12
  }
});
