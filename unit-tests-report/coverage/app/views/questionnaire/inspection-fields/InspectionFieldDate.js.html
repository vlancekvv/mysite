import React from 'react';
import PropTypes from 'prop-types';
import DatePicker from 'react-native-datepicker'
import { StyleSheet, View } from 'react-native';
 
import { getInspectionField } from '../InspectionField';
 
/**
 * Inspection field date component
 */
export class InspectionFieldDate extends React.PureComponent {
 
  /**
   * Should open date picker modal if props have been updated
   */
  componentDidUpdate(prevProps) {
    return !prevProps.isEditing && this.props.isEditing && this.datePicker.onPressDate();
  }
 
  /**
   * Schedules a inspection field date should be rendered
   */
  render() {
    const { value, editingHandler, confirmHandler, cancel, renderDefaultValue } = this.props;
    return (
      <View style={styles.dateContainer}>
        {renderDefaultValue()}
        <DatePicker
          ref={datePicker => this.datePicker = datePicker}
          style={{width: '100%'}}
          format="ddd, D. MMM YYYY"
          placeholder="Bittle eingeben"
          confirmBtnText="Confirm"
          cancelBtnText="Abbrechen"
          customStyles={styles}
          date={value}
          showIcon={false}
          onPressMask={() => {
            this.datePicker.onPressCancel();
            cancel();
          }}
          onDateChange={date => editingHandler(date, confirmHandler)}
        />
      </View>
    );
  }
}
 
InspectionFieldDate.propTypes = {
  value: PropTypes.string,
  editingHandler: PropTypes.func,
  confirmHandler: PropTypes.func,
  startHandler: PropTypes.func,
  renderDefaultValue: PropTypes.func
};
 
const styles = StyleSheet.create({
  dateContainer: {
    flex: 1,
    width: '100%',
    justifyContent: 'center',
    alignItems: 'center',
    borderWidth: 0.25
  },
  dateInput: {
    borderWidth: 0,
  },
  dateText: {
    fontSize: 18
  },
  placeholderText: {
    fontSize: 18
  },
  btnTextConfirm: {
    color: '#007AFF'
  }
});
 
export default getInspectionField(InspectionFieldDate);
