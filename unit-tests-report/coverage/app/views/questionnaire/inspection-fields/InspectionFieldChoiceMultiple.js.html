import React from 'react';
import PropTypes from 'prop-types';
import { StyleSheet, Text, TouchableOpacity } from 'react-native';
 
import { getInspectionField } from '../InspectionField';
import { modalStyle } from '../../../shared/styles';
import { EditFieldModal } from './EditFieldModal';
 
/**
 * Inspection field multiple choice component
 */
export function InspectionFieldChoiceMultiple({value, editingHandler, choices, startHandler, selectedValue, ...props}) {
 
  selectedValue = selectedValue || [];
 
  const activeValue = selectedValue.length ? selectedValue : (props.defaultValue || []);
 
  const editedValue = value || [];
 
  value = value.length ? value : activeValue;
 
  const renderDefaultValue = props.defaultValue && selectedValue.length ?
    props.defaultValue.map((_value, key) => <Text style={styles.defaultValue} key={key}>{_value}</Text>) :
    null;
 
  const choicesList = choices.map((choice, key) => {
    const isAddedChoice = editedValue.find(_value => _value === choice);
      return (
        <TouchableOpacity
          key={key}
          style={modalStyle.modalClick}
          onPress={() => editingHandler(isAddedChoice ? editedValue.filter(_value => _value !== choice) :
            editedValue.concat([choice])
          )}>
          <Text style={{...modalStyle.modalText, color: isAddedChoice ? '#007AFF' : 'black'}}>
            {choice}
          </Text>
        </TouchableOpacity>
      )
    }
  );
 
  return (
    <TouchableOpacity style={styles.choiceContainer} onPress={startHandler}>
      {renderDefaultValue}
      {
        value.length ?
          value.map((value, key) => <Text key={key} style={styles.text}>{value}</Text>) :
          <Text style={{...styles.text, color: '#c7c7c7'}}>Mehrfachauswahl</Text>
      }
      {props.isEditing && <EditFieldModal renderContent={choicesList} {...props} />}
    </TouchableOpacity>
  );
}
 
InspectionFieldChoiceMultiple.propTypes = {
  value: PropTypes.arrayOf(PropTypes.string),
  selectedValue: PropTypes.arrayOf(PropTypes.string),
  defaultValue: PropTypes.arrayOf(PropTypes.string),
  editingHandler: PropTypes.func,
  startHandler: PropTypes.func,
  renderDefaultValue: PropTypes.func,
  choices: PropTypes.arrayOf(PropTypes.string),
  isEditing: PropTypes.bool
};
 
InspectionFieldChoiceMultiple.defaultProps = {
  value: [],
  choices: [],
  selectedValue: [],
  isEditing: false
};
 
const styles = StyleSheet.create({
  choiceContainer: {
    flex: 1,
    width: '100%',
    justifyContent: 'center',
    alignItems: 'center'
  },
  text: {
    fontSize: 18
  },
  defaultValue: {
    color: 'grey',
    fontSize: 12,
    textDecorationLine: 'line-through'
  }
});
 
export default getInspectionField(InspectionFieldChoiceMultiple);
