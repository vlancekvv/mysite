import React, {Component} from 'react'
import {
  AppRegistry,
  StyleSheet,
  Text,
  View,
  Alert,
  Image,
  Dimensions,
  Slider,
  ListView,
  TouchableHighlight,
  TouchableWithoutFeedback
} from 'react-native'
import Logger from '../utils/Logger'
import ImageViewStyle from './ImageView.style'
 
export default class ListItemsViewItem extends Component {
 
  constructor(props) {
    super(props);
  }
  renderPosition = () => {
    if (this.props.item.pos){
      if(!this.props.item.pos.lat || !this.props.item.pos.lng){
        return (
          <Text> None </Text>
        )
      }
      else {
        return (
        <Text>{`position:
lat: ${this.props.item.pos.lat}
lng: ${this.props.item.pos.lng}`}</Text>
        )
      }
    }
  }
  renderCategory = () => {
      if(!this.props.item.category){
        return(
          <Text>none category</Text>
        )
      }
      else {
        return (
          <Text>
            {`category: ${this.props.item.category}`}
          </Text>
        )
      }
  }
  render() {
    return (
      <View>
        {
          this.props.item &&
          <View style={ImageViewStyle.imageWrap}>
            <Image
              style={ImageViewStyle.image}
              source={
                {
                  uri: this.props.item.img.thumbnail
                }
              }
            />
            <View style={ImageViewStyle.details}>
              <Text> {`name: ${this.props.item.name}`} </Text>
              {this.renderPosition()}
              {this.renderCategory()}
              <Text> {`orientation: ${this.props.item.orientation}`} </Text>
            </View>
          </View>
        }
      </View>
    )
  }
}
