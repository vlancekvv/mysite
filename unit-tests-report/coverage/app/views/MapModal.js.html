import React, { Component } from 'react';
import { View, StyleSheet } from 'react-native';
import Modal from 'react-native-modal';
import { connect } from 'react-redux';
 
import { MapComponent } from './MapComponent';
import { toggleMap } from '../actions';
 
/**
 * Map modal component
 */
export class MapModal extends Component {
  render() {
    return (
      <View>
        {
          this.props.isShowMap ?
            <Modal onBackdropPress={this.props.toggleMap}
                   isVisible={this.props.isShowMap}
                   style={styles.modal}>
              <View style ={styles.modalContent}>
                <MapComponent orders={this.props.orders} selectedOrder={this.props.order} isModal/>
              </View>
            </Modal>
            :
            null
        }
      </View>
    )
  }
}
 
const styles = StyleSheet.create({
  modalContent: {
    height: '100%',
    width: '100%'
  },
  modal: {
    top: 50,
    left: 0,
    bottom: 50,
    right: 0,
    alignItems: 'flex-start',
    justifyContent: 'flex-start',
    position: 'absolute'
  },
});
 
const mapStateToProps = state => {
  return {
    isShowMap: state.App.isShowMap
  }
};
 
const mapDispatchToProps = dispatch => {
  return {
    toggleMap: () => dispatch(toggleMap())
  }
};
 
export default connect(mapStateToProps, mapDispatchToProps)(MapModal)
