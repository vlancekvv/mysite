import React, {Component} from 'react'
import {
  Text,
  View,
  TextInput,
  TouchableOpacity,
  TouchableWithoutFeedback,
  Alert,
  Image,
  NetInfo,
  Animated,
  Keyboard,
  Platform,
  Dimensions
} from 'react-native'
import StorageService from '../services/StorageService'
import stylesLogin from './Login.style'
import { PermissionsAndroid } from 'react-native'
import { connect } from 'react-redux'
import  { hardLogin, login } from '../actions/backend'
import {setLiveServer, triggerLoader} from '../actions/index'
import global from "../shared/global";
 
const { height } = Dimensions.get('window');
const formMarginBottom = (height - 350)/2;
 
class LoginView extends Component {
 
  constructor(props) {
    super(props);
    this.state = {
      email: '',
      password: '',
      disabled: false,
      gps: true,
      error: false,
      lastLogoClick: 0,
      countOfClicks: 0,
      keyboardHeight: new Animated.Value(formMarginBottom)
    };
  }
 
  componentDidMount() {
    this.checkCamera();
    this.checkAuth();
    this.keyboardDidShowSub = Keyboard.addListener('keyboardDidShow', this.keyboardDidShow);
    this.keyboardDidHideSub = Keyboard.addListener('keyboardDidHide', this.keyboardDidHide);
  };
 
  componentWillUnmount() {
    this.keyboardDidShowSub.remove();
    this.keyboardDidHideSub.remove();
  }
 
  keyboardDidShow = (event) => {
    Animated.parallel([
      Animated.timing(this.state.keyboardHeight, {
        duration: 1,
        toValue: Platform.OS === 'ios' ?
          (event.endCoordinates.height > formMarginBottom ? event.endCoordinates.height : formMarginBottom) :
          10
      })
    ]).start();
  };
 
  keyboardDidHide = () => {
    Animated.parallel([
      Animated.timing(this.state.keyboardHeight, {
        duration: 1,
        toValue: formMarginBottom,
      })
    ]).start();
  };
 
  checkCamera = () => {
      PermissionsAndroid.request(PermissionsAndroid.PERMISSIONS.CAMERA)
      .then(res => {
          if(res !== 'granted') {
              Alert.alert('Oops!', 'We need permission to access your camera')
          }
          return false;
      });
  };
 
  checkAuth = () => {
    StorageService.getUserEmail()
      .then(email => email && this.setState({email}))
  };
 
  handleEmail = (email) => {
    if (email.match(/ap\w+|ap/) && !this.props.useLiveServer) {
      this.setState({email: 'appuser@sprengnetter.de', password: 'Sprengnetter1'})
    } else {
      this.setState({email});
    }
  };
 
  handleLogin = (userName,userPassword) => {
    NetInfo.getConnectionInfo().then((connectionInfo) => {
      StorageService.getUserEmail().then(i => {
        if (i !== userName && connectionInfo.type === 'none') {
          this.setState({error: !this.state.error})
        }
      });
      if (connectionInfo.type === 'none' && userName) {
        this.setLiveServer();
        return this.props.hardLogin(userName)
      }
      if (userName && userPassword) {
        this.setLiveServer();
        return this.props.login(userName, userPassword)
      }
    })
  };
 
  setLiveServer = () => {
    global.useLiveServer = this.props.useLiveServer ? 'PROD' : 'STAGING';
  };
 
  handleLogoClick = () => {
    if (this.props.useLiveServer) {
      this.setState({countOfClicks: ++this.state.countOfClicks});
      if (this.state.countOfClicks === 5) {
        this.props.setLiveServer(false);
        this.setState({lastLogoClick: 0, countOfClicks: 0})
      }
    } else {
      this.props.setLiveServer(true);
    }
  };
 
 
  render() {
    return (
      <View style={stylesLogin.container} accessibilityLabel="loginScreen">
        <Animated.View style={{...stylesLogin.form, bottom: this.state.keyboardHeight}}>
          <TouchableWithoutFeedback onPress={this.handleLogoClick} accessibilityLabel="logo">
            <Image resizeMode="contain" style={stylesLogin.logo} source={require('../img/SprengnetterLogo.png')}/>
          </TouchableWithoutFeedback>
          <View style={stylesLogin.inputContainer}>
            <Text style={stylesLogin.title}>Benutzername</Text>
            <TextInput
              accessibilityLabel="emailInput"
              style={stylesLogin.input}
              value={this.state.email}
              autoCapitalize='none'
              autoCorrect={false}
              onChangeText={this.handleEmail}
              underlineColorAndroid='transparent'/>
          </View>
          <View style={stylesLogin.inputContainer}>
            <Text style={stylesLogin.title}>Passwort</Text>
            <TextInput
              secureTextEntry={true}
              style={stylesLogin.input}
              value={this.state.password}
              autoCorrect={false}
              autoCapitalize='none'
              onChangeText={(password) => this.setState({password})}
              underlineColorAndroid='transparent'/>
          </View>
          {
            this.state.error ?
            <Text style={stylesLogin.error}>Verbindungsproblem</Text>
            : null
          }
          {
            this.props.passwordError ?
            <Text style={stylesLogin.error}>Benutzername / Passwort falsch</Text>
            : null
          }
          {
            !this.props.useLiveServer ?
              <Text>Login über den Testserver</Text>
              : null
          }
          <TouchableOpacity disabled={this.state.disabled}
                            accessibilityLabel="confirmButton"
                            onPress={() => this.handleLogin(this.state.email, this.state.password)}
                            style={stylesLogin.button}>
            <Text style={stylesLogin.titleButton}>Bestätigen</Text>
          </TouchableOpacity>
        </Animated.View>
      </View>
    );
  }
}
 
const mapStateToProps = state => {
  return {
    auth: state.Api.authorized,
    passwordError: state.Api.passwordError,
    useLiveServer: state.App.useLiveServer
  }
};
 
const mapDispatchToProps = dispatch => {
  return {
    login: (userName, userPassword) => dispatch(login(userName, userPassword)),
    hardLogin: (userName) => dispatch(hardLogin(userName)),
    triggerLoader: () => dispatch(triggerLoader()),
    setLiveServer: liveServer => dispatch(setLiveServer(liveServer))
  }
};
 
export default connect(mapStateToProps, mapDispatchToProps)(LoginView)
