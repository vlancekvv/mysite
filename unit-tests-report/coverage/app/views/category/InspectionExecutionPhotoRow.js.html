import React, { Component } from 'react'
import {View, StyleSheet, Dimensions, Image, TouchableOpacity} from 'react-native'
import { connect } from 'react-redux'
import { setCurrentPhoto, setNeededCategory } from '../../actions'
 
class InspectionExecutionPhotoRow extends Component {
  render() {
    const category = this.props.item.category;
    const TABLET = this.props.tablet;
    const CATEGORY = (category && category !== 'Other');
    const tabletTale = (TABLET && CATEGORY) ? 'TabletBorder' :
    ((TABLET && !CATEGORY) ?  'Tablet' : ((!TABLET && CATEGORY) ? 'Border' : ''));
    return (
      <View style={styles[`imageWrap${tabletTale}`]}>
        <TouchableOpacity
          style={styles[`imageWrap${tabletTale}`]}
          onPress={() => {
            this.props.setNeededCategory(category, this.props.item);
            this.props.setCurrentPhoto(this.props.item);
            this.props.toggleModal();
          }}>
        <Image
          resizeMode='cover'
          style={styles.image}
          source={{
          uri: this.props.item.img.thumbnail}}/>
        </TouchableOpacity>
      </View>
    )
  }
}
 
const styles = StyleSheet.create({
  image: {
    flex: 1,
    alignSelf: 'stretch',
  },
  imageWrap: {
    borderColor: 'gray',
    borderWidth: 1,
    width: (Dimensions.get('window').width / 2),
    height: (Dimensions.get('window').width / 2),
    flexWrap: 'wrap',
    flexDirection: 'column',
    justifyContent: 'space-between',
    alignSelf: 'flex-start',
  },
  imageWrapBorder: {
    borderColor: 'blue',
    borderWidth: 1.5,
    width: (Dimensions.get('window').width / 2),
    height: (Dimensions.get('window').width / 2),
    flexWrap: 'wrap',
    flexDirection: 'column',
    justifyContent: 'space-between',
    alignSelf: 'flex-start',
    paddingBottom: 1,
    paddingRight: 1,
    zIndex: 5,
  },
  imageWrapTablet: {
    borderColor: 'gray',
    borderWidth: 1,
    backgroundColor: '#fff',
    width: (Dimensions.get('window').width / 6),
    height: (Dimensions.get('window').width / 6),
    flexWrap: 'wrap',
    flexDirection: 'column',
    justifyContent: 'space-between',
    alignSelf: 'flex-start',
  },
  imageWrapTabletBorder: {
    borderColor: 'blue',
    borderWidth: 1.5,
    backgroundColor: '#fff',
    width: (Dimensions.get('window').width / 6),
    height: (Dimensions.get('window').width / 6),
    flexWrap: 'wrap',
    flexDirection: 'column',
    justifyContent: 'space-between',
    alignSelf: 'flex-start',
    paddingBottom: 1,
    paddingRight: 1,
    zIndex: 5,
  },
});
 
const mapStateToProps = state => {
  return {
    tablet: state.App.isTablet
  }
};
 
const mapDispatchToProps = dispatch => {
  return {
    setCurrentPhoto: (item) => dispatch(setCurrentPhoto(item)),
    setNeededCategory: (name,item) => dispatch(setNeededCategory(name,item))
  }
};
 
export default connect(mapStateToProps, mapDispatchToProps)(InspectionExecutionPhotoRow)
