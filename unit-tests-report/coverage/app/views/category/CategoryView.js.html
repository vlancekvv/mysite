import React, { Component } from "react";
import { View, StyleSheet, BackHandler } from 'react-native';
import ExecutionCategoryView from "./ExecutionCategoryView";
import InspectionExecutionScreen from "./InspectionExecutionScreen";
import { connect } from "react-redux";
import Icon from "react-native-vector-icons/MaterialCommunityIcons";
import MaterialIcon from "react-native-vector-icons/MaterialIcons";
import "moment/locale/de";
import TabNavigator from 'react-native-tab-navigator';
import { NavigationActions, StackActions } from 'react-navigation';
 
import { toggleMap } from '../../actions';
 
export class CategoryView extends Component {
 
  componentDidMount() {
    BackHandler.addEventListener('hardwareBackPress', this.handleBackPress);
  };
 
  componentWillUnmount() {
    BackHandler.removeEventListener('hardwareBackPress', this.handleBackPress);
  }
 
  handleBackPress = () => {
    const resetAction = StackActions.reset({
      index: 0,
      actions: [NavigationActions.navigate({ routeName: 'Home' })],
    });
    this.props.navigation.dispatch(resetAction);
    return true;
  };
 
  render() {
    const iconSize = this.props.tablet ? 30 : 25,
      titleFontSize = this.props.tablet ? 14 : 12;
    return (
      <View style={styles.container}>
        <TabNavigator tabBarStyle={{height: 55}}>
          <TabNavigator.Item
            selected={true}
            tabStyle={{display: 'none'}}>
            {this.props.orders.length ? <ExecutionCategoryView navigation={this.props.navigation}/> : <View/>}
          </TabNavigator.Item>
          <TabNavigator.Item
            renderIcon={() => <Icon name={ 'menu' } size={ iconSize }/>}
            renderSelectedIcon={() => <Icon name={ 'menu' } size={ iconSize } color={ '#3c5296' }/>}
            title='Navigation'
            tabStyle={{display: 'none'}}
            titleStyle={{fontSize: titleFontSize}}
            selectedTitleStyle={{color: '#3c5296'}}>
          </TabNavigator.Item>
          <TabNavigator.Item
            renderIcon={() => <Icon name={ 'image-area' } size={ iconSize }/>}
            renderSelectedIcon={() => <Icon name={ 'image-area' } size={ iconSize } color={ '#3c5296' }/>}
            badgeText={this.props.totalFile}
            title='Fotos'
            titleStyle={{fontSize: titleFontSize}}
            onPress={ () => this.props.navigation.navigate('InspectionExecutionScreen') }
            selectedTitleStyle={{color: '#3c5296'}}>
          </TabNavigator.Item>
          <TabNavigator.Item
            renderIcon={() => <Icon name={ 'camera' } size={ iconSize }/>}
            renderSelectedIcon={() => <Icon name={ 'camera' } size={ iconSize } color={ '#3c5296' }/>}
            title='Kamera'
            titleStyle={{fontSize: titleFontSize}}
            onPress={ () => this.props.navigation.navigate("Capture", { title: this.props.orderName, id: this.props.order }) }
            selectedTitleStyle={{color: '#3c5296'}}>
          </TabNavigator.Item>
          <TabNavigator.Item
            renderIcon={() => <Icon name={ 'map' } size={ iconSize }/>}
            renderSelectedIcon={() => <Icon name={ 'map' } size={ iconSize } color={ '#3c5296' }/>}
            selectedTitleStyle={{color: '#3c5296'}}
            titleStyle={{fontSize: titleFontSize}}
            onPress={this.props.toggleMap}
            title='Karte'>
          </TabNavigator.Item>
          <TabNavigator.Item
            renderIcon={() => <MaterialIcon name={ 'warning' } size={ iconSize }/>}
            tabStyle={{display: 'none'}}
            renderSelectedIcon={() => <MaterialIcon name={ 'warning' } size={ iconSize } color={ '#3c5296' }/>}
            selectedTitleStyle={{color: '#3c5296'}}
            titleStyle={{fontSize: titleFontSize}}
            title='Validierung'>
          </TabNavigator.Item>
        </TabNavigator>
      </View>
    );
  }
}
const styles = StyleSheet.create({
  container: {
    backgroundColor: "#fff",
    flex: 1,
    flexDirection: "column"
  }
});
 
const mapStateToProps = state => {
  return {
    totalFile: state.Home.totalFile,
    tablet: state.App.isTablet,
    orders: state.Api.orders,
    orderName: state.App.currentOrderName,
    order: state.App.currentOrder
  };
};
 
const mapDispatchToProps = dispatch => {
  return {
    toggleMap: () => dispatch(toggleMap())
  }
};
 
export default connect(mapStateToProps, mapDispatchToProps)(CategoryView);
