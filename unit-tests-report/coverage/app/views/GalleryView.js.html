import React, {Component} from 'react'
import {
  StyleSheet,
  View,
  Image,
  Dimensions,
  TouchableOpacity, Alert
} from 'react-native'
import StorageService from '../services/StorageService'
import Icon from 'react-native-vector-icons/MaterialCommunityIcons'
import { connect } from 'react-redux'
import { changeCurrentPhoto } from '../actions/index'
 
class GalleryView extends Component {
 
  constructor(props) {
    super(props);
    this.state = {
      portrait: false,
    }
  }
 
  componentDidMount = () => {
		if( this.props.orientation === 'PORTRAIT' ) {
			this.setState ({portrait: true})
		}
		else { this.setState ({portrait: false}) }
    this.handleBorder(this.props.item.img)
	};
 
  handleBorder = () => {
    this.setState({editMode: !this.state.editMode})
  };
 
  toPostCapture = () => {
    const item = this.props.item;
    StorageService.getOriginal(item.id).then((original) => {
      const payload = { ...item };
      payload.img = { ...payload.img, original };
      this.props.navigate.navigate('PostCaptureView', { payload, title: this.props.orderName, databaseId: item.id, isAddedPhoto: true })
    })
      .catch(() => Alert.alert('Cannot get photo from storage'))
  };
 
  render() {
    const PORTRAIT = this.state.portrait;
    return (
        <TouchableOpacity style={ !this.props.tablet ? GalleryViewStyle.imageWrap
          : ((this.props.tablet && PORTRAIT)? GalleryViewStyle.imageWrapPortrait
            : GalleryViewStyle.imageWrapTablet)}
                          onPress={() => {
                            this.handleBorder(this.props.item.img);
                            this.toPostCapture();
                            this.props.changeCurrentPhoto(this.props.item.id);
                          }}>
          <Image resizeMode='cover'
                 style={GalleryViewStyle.image}
                 source={{uri: this.props.item.img.thumbnail}}/>
          <View style={GalleryViewStyle.delete}>
            <TouchableOpacity onPress={() => this.props.deletePhoto(this.props.item)}>
              <Icon
                name="close"
                size={30}
                color="red"/>
            </TouchableOpacity>
          </View>
        </TouchableOpacity>
    );
  }
}
const GalleryViewStyle = StyleSheet.create({
  image: {
    flex: 1,
  },
  imageWrapPortrait: {
    backgroundColor: '#fff',
    width: (Dimensions.get('window').width / 7),
    height: (Dimensions.get('window').width / 7),
    flexWrap: 'wrap',
    flexDirection: 'column',
    alignSelf: 'flex-start'
  },
  imageWrapTablet: {
    backgroundColor: '#fff',
    width: (Dimensions.get('window').width / 7),
    height: (Dimensions.get('window').width / 7),
    flexWrap: 'wrap',
    flexDirection: 'column',
    alignSelf: 'flex-start'
  },
  imageWrap: {
    backgroundColor: '#fff',
    width: (Dimensions.get('window').width / 2),
    height: (Dimensions.get('window').width / 2),
    flexWrap: 'wrap',
    flexDirection: 'column',
    alignSelf: 'flex-start'
  },
  delete: {
    display: 'flex',
    alignItems: 'center',
    justifyContent: 'center',
    borderWidth: 2,
    borderColor: 'red',
    position: 'absolute',
    top: 0,
    right: 0,
    height: 40,
    width: 40,
    padding: 0,
    zIndex: 2
  }
});
 
const mapStateToProps = state => {
  return {
    tablet: state.App.isTablet,
    orientation: state.App.orientation,
    currentPhoto: state.App.currentPhoto,
    orderName: state.App.currentOrderName,
  }
};
 
const mapDispatchToProps = dispatch => {
  return {
    changeCurrentPhoto : (id) =>dispatch(changeCurrentPhoto(id)),
 
  }
};
 
export default connect(mapStateToProps, mapDispatchToProps)(GalleryView)
