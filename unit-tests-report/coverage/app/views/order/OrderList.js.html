import React, { Component } from 'react'
import {View, Text, ScrollView, StyleSheet, RefreshControl } from 'react-native'
import { connect } from 'react-redux'
import { triggerLoader, changeCurrentOrder } from '../../actions'
import {
  getUpdatedOrders,
  toggleRefreshing,
  getLastSynchronizationTime,
  setManualUpdatingFlag,
  handleSynchronizationError
} from '../../actions/backend';
import RenderOrder from './RenderOrder'
 
class OrderList extends Component {
 
  componentDidMount = () => {
    this.props.getLastSynchronizationTime();
  };
 
  _renderOrder = () => {
    return this.props.orders
      .filter(order => order.status === 30 || order.status === 35)
      .sort((previousOrder, nextOrder) =>
        previousOrder[this.getFieldForSorting(previousOrder)] - nextOrder[this.getFieldForSorting(nextOrder)])
      .map((order, key) => <RenderOrder navigation={this.props.navigation}
                                        isArchiveOrder={false}
                                        item={order}
                                        key={key}/>)
  };
 
  getFieldForSorting = order => order.inspectionModelLabel === 'Innen' && order.appointment > new Date().getTime() ?
    'appointment' : 'dueTimestamp';
 
  onRefresh = () => {
    this.props.toggleRefreshing();
    this.props.getUpdatedOrders()
      .catch(error => this.props.handleSynchronizationError({...error, message: 'Verbindungsproblemâ€¦'}))
      .then(() => {
        this.props.setManualUpdatingFlag();
        this.props.toggleRefreshing();
      })
  };
 
  render() {
    const orders = this._renderOrder();
    return (
      <View style={styles.container}>
        <ScrollView refreshControl={
          <RefreshControl refreshing={this.props.refreshing}
                          onRefresh={this.onRefresh}/>
        } style={{flex: 1}}>
          {
          !orders.length ?
            <View style={styles.textContainer}>
              <Text style={{fontSize: 18}}>
                {
                  this.props.refreshing ?
                    'Empfange Auftragsliste vom Server...' :
                    'Es liegen keine AuftrÃ¤ge vor'
                }
              </Text>
            </View>
            : orders
          }
        </ScrollView>
      </View>
    )
  }
}
 
const styles = StyleSheet.create({
  container: {
    flex: 1,
    width: '100%'
  },
  textContainer: {
    width: '100%',
    justifyContent: 'center',
    alignItems: 'center',
    marginTop: 50
  }
});
 
const mapStateToProps = state => {
  return {
    tablet: state.App.isTablet,
    order: state.App.currentOrder,
    orders: state.Api.searchedOrders ? state.Api.searchedOrders : state.Api.orders,
    refreshing: state.Api.refreshingOrders
  }
};
 
const mapDispatchToProps = dispatch => {
  return {
    triggerLoader: () => dispatch(triggerLoader()),
    changeCurrentOrder: (order) => dispatch(changeCurrentOrder(order)),
    getUpdatedOrders: () => dispatch(getUpdatedOrders()),
    toggleRefreshing: () => dispatch(toggleRefreshing()),
    getLastSynchronizationTime: () => dispatch(getLastSynchronizationTime()),
    setManualUpdatingFlag: () => dispatch(setManualUpdatingFlag()),
    handleSynchronizationError: error => dispatch(handleSynchronizationError(error))
  }
};
 
export default connect(mapStateToProps, mapDispatchToProps)(OrderList)
