import React, { Component } from 'react';
import { View, StyleSheet } from 'react-native';
import { connect } from 'react-redux';
 
import OrderList from './OrderList';
import Menu from './Menu';
import UpdatingOrdersInfo from './UpdatingOrdersInfo';
import ModalMap from '../MapModal';
import { hardLogin, login, submitOrder } from '../../actions/backend';
import { resetCurrentOrder } from '../../actions';
import { MapComponent } from '../MapComponent';
 
 class OrderView extends Component {
 
  constructor(props){
    super(props);
    this.state = {
      renderRow: false,
      active: false
    };
    this.castCurrent();
    this.handleRow();
  }
 
  castCurrent = () => {
    if (this.props.currentOrder){
      this.props.resetCurrentOrder()
    }
  };
 
  handleRow = () => {
    if (this.props.tablet) {
      this.setState({renderRow: true})
    }
  };
 
  handleRowTablet = () => {
    this.setState({renderRow: !this.state.renderRow})
  };
 
  render() {
    const tablet = this.props.tablet;
    const { orders, order } = this.props;
    return (
      <View style={styles.container} accessibilityLabel="ordersScreen">
        <View style={!tablet ? styles.listContainer : styles.listContainerTablet}>
          {
            tablet ?
              <View style={styles.mapContainer}>
                <MapComponent orders={orders} selectedOrder={order}/>
              </View> :
              null
          }
          <OrderList navigation={this.props.navigation} />
          <ModalMap orders={orders} order={order} />
        </View>
        <Menu navigation={this.props.navigation} />
        <UpdatingOrdersInfo />
      </View>
    )
  }
}
 
const styles = StyleSheet.create({
  container: {
    flex: 1,
    backgroundColor: '#fff',
    flexDirection: 'column',
    position: 'relative'
  },
  listContainer: {
    flex: 1,
    flexDirection: 'row',
  },
  listContainerTablet: {
    flex: 1,
    flexDirection: 'row',
  },
  mapContainer: {
    flex: 1,
    width: '100%'
  }
});
 
const mapStateToProps = state => {
  return {
    tablet: state.App.isTablet,
    currentOrder: state.App.currentOrder,
    orderName: state.App.currentOrderName,
    savedOrders: state.App.savedOrders,
    auth: state.Api.authorized,
    orders: state.Api.orders,
    order: state.App.currentOrderObj
  }
};
 
const mapDispatchToProps = dispatch => {
  return {
    submitOrder: (currentOrder, callback) => dispatch(submitOrder(currentOrder, callback)),
    resetCurrentOrder: () => dispatch(resetCurrentOrder()),
    login: (userName, userPassword) => dispatch(login(userName, userPassword)),
    hardLogin: (userName) => hardLogin(userName)
  }
};
 
export default connect(mapStateToProps, mapDispatchToProps)(OrderView)
