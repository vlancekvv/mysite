export function getLiveServer() {
  return dispatch => StorageService.getLiveServer()
    .then(liveServer => {
      global.useLiveServer = liveServer === false ? 'STAGING' : 'PROD';
      dispatch(saveLiveServerInState(liveServer !== false))
    })
}
 
export function setLiveServer(liveServer) {

    .then(() => dispatch(saveLiveServerInState(liveServer)))
}
