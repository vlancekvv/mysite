import React, { Component } from 'react'
import {
  View,
  Alert,
  Text,
  ScrollView
} from 'react-native'
import { connect } from 'react-redux'
import StorageService from '../services/StorageService'
import GalleryView from './GalleryView'
import ImageListViewStyle from './ImageListView.style'
import {reloadGallery, changeCurrentPhoto, deleteItem, deletePhoto} from '../actions/index'
 
class ListItemsView extends Component {
 
  constructor(props) {
    super(props);
    this.state = {
      portrait: false,
    };
  }
 
  componentDidMount = () => {
    this.setState({portrait: this.props.orientation === 'PORTRAIT'})
	};
 
  _renderPhotoRow = () => {
    return this.props.photos
      .sort((prevPhoto, nextPhoto) => prevPhoto.dateInMilliseconds - nextPhoto.dateInMilliseconds)
      .map((photoObj, key ) =>
         <GalleryView
          key={key}
          navigate = {this.props.navigation}
          deletePhoto={this.deletePhoto}
          item={photoObj}/>
        );
  };
 
  deletePhoto = (item) => {
    Alert.alert('', 'Sind Sie sicher, dass Sie dieses Foto löschen möchten?', [
      {
        text: 'Abbrechen',
        onPress: () => console.log('Cancel Pressed'),
        style: 'cancel'
      }, {
        text: 'Löschen',
        onPress: () => {
          StorageService.deleteItem(item.id);
          this.props.deleteItem(item);
          this.props.deletePhoto(item);
        }
      }
    ], {cancelable: false})
  };
 
  render() {
    const PORTRAIT = this.state.portrait;
    const TABLET = this.props.tablet;
    return (
      <View style={ImageListViewStyle.container}>
        {this.props.photos.length ?
            <ScrollView
              horizontal={TABLET && PORTRAIT}
              contentContainerStyle={!this.props.tablet ?
                ImageListViewStyle.listView :
                ((this.props.tablet && PORTRAIT) ?
                  ImageListViewStyle.listViewTabletPortrait :
                  ImageListViewStyle.listViewTablet)
              }>
              {this._renderPhotoRow()}
            </ScrollView> :
          <View style={ImageListViewStyle.info}>
            <Text style={ImageListViewStyle.infotitle}>Hier sehen Sie Ihre Bilder.</Text>
          </View>
        }
      </View>
    );
  }
}
 
const mapStateToProps = state => {
  return {
    tablet: state.App.isTablet,
    orientation: state.App.orientation,
    photos: state.App.dataPhotos,
  }
};
 
const mapDispatchToProps = dispatch => {
  return {
    reloadGallery: (data) => dispatch(reloadGallery(data)),
    changeCurrentPhoto : (id) =>dispatch(changeCurrentPhoto(id)),
    deleteItem:(id) => dispatch(deleteItem(id)),
    deletePhoto: photo => dispatch(deletePhoto(photo))
  }
};
 
export default connect(mapStateToProps, mapDispatchToProps)(ListItemsView)
