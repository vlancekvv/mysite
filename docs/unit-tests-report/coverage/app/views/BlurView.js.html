import React, {Component} from 'react';
import { View, BackHandler } from 'react-native';
import {connect} from 'react-redux'
import _ from 'lodash'
 
import BlurEditor from '../../packages/photo-module/components/BlurEditor'
 
class BlurView extends Component {
 
  componentDidMount = () => {
    BackHandler.addEventListener('hardwareBackPress', this.handleBackPress);
  };
 
  componentWillUnmount() {
    BackHandler.removeEventListener('hardwareBackPress', this.handleBackPress);
  }
 
  handleBackPress = () => {
    this.props.navigation.navigate('CategoryView', {title: this.props.orderName, id: this.props.currentOrder});
    return true;
  };
 
  saveAndGoBack = (payload) => {
    payload.img.original = `data:image/png;base64,${payload.img.original}`;
    this.props.navigation.navigate(
      'PostCaptureView',
      {
        payload,
        oldPayload: this.props.navigation.state.params.payload,
        title: this.props.orderName
      })
  };
 
  render = () => {
    const { rotate90deg } = { ...this.props.navigation.state.params };
    const data = _.cloneDeep(this.props.navigation.state.params.payload);
    return (
        <View style={{top: 0, bottom: 0, left: 0, right: 0, position: 'absolute'}}>
            <BlurEditor data = { data } rotate90deg = { rotate90deg } save = {this.saveAndGoBack}/>
        </View>
    );
  }
 
}
 
const mapStateToProps = state => {
  return {
    orderName: state.App.currentOrderName,
    currentOrder: state.App.currentOrder
  }
};
 
export default connect(mapStateToProps)(BlurView)
