import React, { Component } from 'react'
import {View, StyleSheet, Text} from 'react-native'
import Modal from "react-native-modal";
import DeviceInfo from 'react-native-device-info';
import { connect } from 'react-redux';
 
import config from '../utils/config';
import global from "../shared/global";
import {toggleAppInfo} from "../actions";
import {Header} from "react-navigation";
 
export class AppInfo extends Component {
  render() {
    return (
      <Modal animationInTiming={1}
             animationOutTiming={1}
             onBackdropPress={this.props.toggleAppInfo}
             backdropOpacity={0}
             isVisible={this.props.isShowAppInfo}
             style={styles.menu}>
        <View style={styles.triangle}/>
        <View style={styles.modalContent}>
          <View style={styles.modalClick}>
            <Text style={styles.text}>App Version</Text>
            <Text style={styles.text}>{`${DeviceInfo.getVersion()}(${DeviceInfo.getBuildNumber()})`}</Text>
          </View>
          <View style={[styles.modalClick, this.props.useLiveServer ? styles.lastModalClick : {}]}>
            <Text style={styles.text}>Benutzer</Text>
            <Text style={styles.text}>{this.props.userName}</Text>
          </View>
          {
            !this.props.useLiveServer ?
              <View style={[styles.modalClick, styles.lastModalClick]}>
                <Text style={styles.text}>Server</Text>
                <Text style={styles.text}>{config[`${global.useLiveServer}_API_BASE`]}</Text>
              </View> :
              null
          }
        </View>
      </Modal>
    )
  }
}

    borderBottomColor: '#eaeaea'
  }
});
 
const mapStateToProps = state => {
  return {
    isShowAppInfo: state.App.isShowAppInfo,
    userName: state.Api.userName,
    useLiveServer: state.App.useLiveServer
  }
};
 
const mapDispatchToProps = dispatch => {
  return {
    toggleAppInfo: () => dispatch(toggleAppInfo())
  }
};
 
export default connect(mapStateToProps, mapDispatchToProps)(AppInfo)
