import React, { Component } from 'react';
import { View, StyleSheet, Alert, Text, TouchableOpacity } from 'react-native';
import { connect } from 'react-redux';
import Icon from 'react-native-vector-icons/MaterialCommunityIcons';
import moment from 'moment';
 
import { toggleMap } from '../../actions';
 
class UpdatingOrdersInfo extends Component {
 
  showErrorDetails = () => {
    const error = this.props.synchronizationError;
    if (error.message === 'Netzwerk offline') {
      Alert.alert('WiFi / Mobilfunkverbindung ausgeschaltet');
    } else if (error.message === 'Verbindungsproblem…') {
      Alert.alert('Keine Verbindung')
    } else if (error.status === 504) {
      Alert.alert('Zeitüberschreitung')
    } else {
      Alert.alert(error.message);
    }
  };
 
  render() {
    const { synchronizationError, lastSynchronization, refreshing } = this.props;
    return (
      <View style={styles.container}>
        <TouchableOpacity onPress={this.showErrorDetails}
                          style={styles.toolbar}
                          disabled={!synchronizationError}>
          <View style={styles.textContainer}>
            {
              !refreshing ?
                <Text style={synchronizationError ? {fontSize: 10} : null}>Letzte Synchronisierung</Text> :
                <Text>Empfange Auftragsliste vom Server...</Text>
            }
            {
              !refreshing ?
                <Text style={synchronizationError ? {fontSize: 10} : null}>
                  {
                    lastSynchronization ?
                      moment(lastSynchronization).format('dd., D. MMM YYYY, kk:mm:ss') :
                      'Die Synchronisation ist noch nicht abgeschlossen'
                  }
                </Text> : null
            }
            { synchronizationError ? <Text>{synchronizationError.message}</Text> : null }
          </View>
        </TouchableOpacity>
        {!this.props.isTablet && (<TouchableOpacity style={styles.iconButton} onPress={this.props.toggleMap}>
          <Icon name={ 'map' } size={ 30 } color={ '#3c5296' } style={styles.icon}/>
          <Text style={{color: '#3c5296'}}>Karte</Text>
        </TouchableOpacity>
        )}
      </View>
    )
  }
}
 
const styles = StyleSheet.create({
  container: {
    height: 75,
    width: '100%',
    backgroundColor: '#fff',
    position: 'relative'
  },
  toolbar: {
    height: 75,
    width: '100%',
    justifyContent: 'center',
    alignItems: 'center',
    borderWidth: 1,
    borderColor: '#bababa',
    zIndex: 10,
    backgroundColor: 'white'
  },
  textContainer: {
    width: '100%',
    justifyContent: 'center',
    alignItems: 'center',
  },
  iconButton: {
    position: 'absolute',
    left: 30,
    top: 15,
    zIndex: 10
  },
  icon: {
 
  }
});
 
const mapStateToProps = state => {
  return {
    isTablet: state.App.isTablet,
    lastSynchronization: state.Api.lastSynchronization,
    synchronizationError: state.Api.synchronizationError,
    refreshing: state.Api.refreshingOrders
  }
};
 
const mapDispatchToProps = dispatch => {
  return {
    toggleMap: () => dispatch(toggleMap())
  }
};
 
export default connect(mapStateToProps, mapDispatchToProps)(UpdatingOrdersInfo)
