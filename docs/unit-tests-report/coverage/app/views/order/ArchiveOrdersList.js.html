import React, { Component } from 'react';
import { View, Dimensions, Text, ScrollView, StyleSheet } from 'react-native';
import { connect } from 'react-redux';
import _ from 'lodash';
 
import RenderOrder from './RenderOrder';
 
export class ArchiveOrdersList extends Component {
 
  renderOrders = () => {
    return _.sortBy(this.props.orders.filter(order => order.status !== 30 && order.status !== 35), ['status', 'goVID'])
      .sort((previousOrder, nextOrder) => {
        if (previousOrder.status === 50 && nextOrder.status === 50) {
          return -(previousOrder.expiration - nextOrder.expiration)
        }
      })
      .map((order, key) =>
        <RenderOrder
          navigate={this.props.navigation}
          item={order}
          isArchiveOrder={true}
          key={key}
        />
      )
  };
 
  render() {
    const archiveOrders = this.renderOrders();
    return (
      <View style={styles.container} accessibilityLabel="archiveScreen">
        { archiveOrders.length <= 0
          ?
          <View style={styles.textContainer}>
            <Text style={{fontSize:18}}>Es liegen keine Auftr√§ge vor</Text>
          </View>
          :
          <ScrollView>{archiveOrders}</ScrollView>
        }
      </View>
    )
  }
}
 
const styles = StyleSheet.create({
  container: {
    flex: 1,
    backgroundColor: 'white'
  },
  textContainer: {
    width: (Dimensions.get('window').width),
    flex: 1,
    justifyContent: 'center',
    alignItems:'center',
    textAlign: 'center'
  }
});
 
const mapStateToProps = state => {
  return {
    orders: state.Api.orders
  }
};
 
export default connect(mapStateToProps)(ArchiveOrdersList)
