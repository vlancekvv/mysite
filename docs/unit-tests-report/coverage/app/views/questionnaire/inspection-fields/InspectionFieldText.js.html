import React from 'react';
import PropTypes from 'prop-types';
import { TextInput, StyleSheet, View } from 'react-native';
 
import { getInspectionField } from '../InspectionField';
 
/**
 * Inspection field text component
 */
export class InspectionFieldText extends React.PureComponent {
 
  /**
   * Should toggle focus and help buttons if props have been updated
   */
  componentDidUpdate(prevProps) {
     if (!prevProps.isEditing && this.props.isEditing){
       this.input.focus();
       this.props.toggleButtons()
     } else if (prevProps.isEditing && !this.props.isEditing) {
       this.props.toggleButtons();
       this.input.blur();
     }
  }
 
  /**
   * Schedules a inspection field text should be rendered
   */
  render() {
    const { isMultiLine, value, confirmHandler, startHandler, editingHandler, ...props } = this.props;
    return (
      <View style={{...styles.inputContainer, width: isMultiLine ? '95%' : '100%', alignItems: isMultiLine ? 'flex-start' : 'center'}}>
        {props.renderDefaultValue()}
        <TextInput
          ref={input => this.input = input}
          style={{...styles.input, minHeight: isMultiLine ? 60 : 0, textAlign: isMultiLine ? 'left' : 'center'}}
          keyboardType={props.isNumber ? 'number-pad' : 'default'}
          placeholder="Bittle eingeben"
          placeholderTextColor={"#c7c7c7"}
          multiline={isMultiLine}
          maxLength={480}
          value={value || ''}
          onBlur={confirmHandler}
          onFocus={startHandler}
          onChangeText={(text) => editingHandler(text)}
        />
      </View>
    );
  }
}
 
InspectionFieldText.propTypes = {
  value: PropTypes.string,
  editingHandler: PropTypes.func,
  confirmHandler: PropTypes.func,
  startHandler: PropTypes.func,
  renderDefaultValue: PropTypes.func,
  isNumber: PropTypes.bool,
  isMultiLine: PropTypes.bool
};
 
InspectionFieldText.defaultProps = {
  isNumber: false,
  isMultiLine: false
};
 
const styles = StyleSheet.create({
  inputContainer: {
    flex: 1,
    justifyContent: 'center',
    alignItems: 'center',
    flexDirection: 'column',
    borderWidth: 0.25,
    paddingHorizontal: 10
  },
  input: {
    fontSize: 18,
    width: '100%',
    margin: 'auto'
  }
});
 
export default getInspectionField(InspectionFieldText);
