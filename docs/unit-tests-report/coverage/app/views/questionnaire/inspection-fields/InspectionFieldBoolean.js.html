import React from 'react';
import PropTypes from 'prop-types';
import { Switch } from 'react-native-switch';
 
import { getInspectionField } from '../InspectionField';
 
/**
 * Inspection field boolean component
 */
export function InspectionFieldBoolean({value, editingHandler, confirmHandler}) {
  return (
    <Switch
      circleSize={25}
      barHeight={30}
      switchLeftPx={2.5}
      switchRightPx={2.5}
      circleBorderWidth={0}
      backgroundActive={'#34c631'}
      backgroundInactive={'#ededed'}
      value={value === 'Ja'}
      onValueChange={() => editingHandler(value === 'Ja' ? 'Nein' : 'Ja', confirmHandler)}
    />
  );
}
 
InspectionFieldBoolean.propTypes = {
  value: PropTypes.string,
  editingHandler: PropTypes.func,
  confirmHandler: PropTypes.func
};
 
export default getInspectionField(InspectionFieldBoolean);
