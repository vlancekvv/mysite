import React, { Component } from 'react';
import { View, StyleSheet, Text } from 'react-native';
import { connect } from 'react-redux';
 
import {
  resetNextAttribute,
  selectAttribute,
  selectNextAttribute,
  selectPrevAttribute,
  toggleButtons,
  updateAttribute
} from './actions/questionnaire.actions';
 
/**
 * Should return changed field component
 * @param FieldComponent
 * @return Changed field component
 */
export function getInspectionField(FieldComponent) {
 
  /**
   * Inspection field component
   */
  class InspectionField extends Component {
 
    constructor(props) {
      super(props);
      this.confirmHandler = this.confirmHandler.bind(this);
      this.cancelHandler = this.cancelHandler.bind(this);
      this.getPropsForFieldComponent = this.getPropsForFieldComponent.bind(this);
    }
 
    /**
     * Check whether the component should be update or not.
     */
    shouldComponentUpdate(nextProps, nextState) {
      const { field } = this.props;
      if (this.state !== nextState) return true;
      else if (field.selectedValue !== nextProps.field.selectedValue) return true;
      else return field.id === this.props.fieldId || field.id === nextProps.nextFieldId;
    };
 
    /**
     * Should update state after selecting next field
     */
    componentDidUpdate(prevProps) {
      if ((prevProps.nextFieldId !== this.props.nextFieldId) && this.props.nextFieldId) {
        const isNextField = this.props.field.id === this.props.nextFieldId;
        this.setState({editing: isNextField});
        return isNextField && this.props.selectAttribute(this.props.field);
      }
    }
 
    /**
     * Should handle start editing event
     */
    startEditingHandler() {
      this.props.resetNextAttribute();
      this.props.selectAttribute(this.props.field);
      this.setState({editing: true})
    };
 
    /**
     * Should handle editing event
     */
    editingHandler(editedValue, callback) {
      this.setState({editedValue}, callback);
    };
 
    /**
     * Should handle confirm editing event
     */
    confirmHandler() {
      this.props.updateAttribute(this.props.field.id, this.editedValue);
      this.cancelHandler();
    };
 
    /**
     * Should handle reset value event
     */
    resetToDefaultHandler() {
      this.props.updateAttribute(this.props.field.id, undefined);
      this.setState({editing: false, editedValue: undefined});
    };
 
    /**
     * Should handle cancel editing event
     */
    cancelHandler() {
      this.props.resetNextAttribute();
      this.setState({editing: false, editedValue: undefined});
    };
 
    /**
     * Should select next field and confirm editing
     */
    selectNextField() {
      this.confirmHandler();
      this.props.selectNextAttribute();
    };
 
    /**
     * Should select previous field and confirm editing
     */
    selectPrevField() {
      this.confirmHandler();
      this.props.selectPrevAttribute();
    };
 
    /**
     * @returns {boolean} Editing flag
     */
    get isEditing() {
      return this.state && this.state.editing;
    }
 
    /**
     * @returns {string|Array<string>} Edited value or active value
     */
    get editedValue() {
      return this.state && this.state.editedValue !== undefined ? this.state.editedValue : this.activeValue;
    }
 
    /**
     * @returns {string|Array<string>|undefined} Selected value or default value
     */
    get activeValue() {
      return this.props.field.selectedValue || this.props.field.defaultValue || undefined;
    }
 
    /**
     * Schedules a default value should be rendered
     */
    renderDefaultValue = () => {
      const { defaultValue, selectedValue } = this.props.field;
      if (defaultValue && selectedValue && defaultValue !== selectedValue) {
        return <Text style={styles.defaultValue}>{defaultValue}</Text>
      }
    };
 
    /**
     * Schedules a props for field component should be returned
     */
    getPropsForFieldComponent(neededProps) {
      return {
        ...neededProps,
        value: this.editedValue,
        defaultValue: this.props.field.defaultValue,
        selectedValue: this.props.field.selectedValue,
        fieldName: this.props.field.name,
        choices: ['select1', 'select2', 'select3', 'select4', 'select5', 'select6'],
        renderDefaultValue: this.renderDefaultValue,
        isEditing: this.isEditing,
        editingHandler: this.editingHandler.bind(this),
        startHandler: this.startEditingHandler.bind(this),
        confirmHandler: this.confirmHandler,
        cancel: this.cancelHandler,
        reset: this.resetToDefaultHandler.bind(this),
        selectNextField: this.selectNextField.bind(this),
        selectPrevField: this.selectPrevField.bind(this),
        toggleButtons: this.props.toggleButtons
      }
    };
 
    /**
     * Schedules a inspection field should be rendered
     */
    render() {
      let { field, ...neededProps } = this.props;
      neededProps = this.getPropsForFieldComponent(neededProps);
      const isMultiLineText = field.fieldType === 'Text Multiline';
      return (
        <View style={{borderBottomColor: 'grey', borderBottomWidth: 0.25}}>
          <View style={styles.container}>
            <View style={styles.fieldContainer}>
              <Text style={styles.text}>{field.name}</Text>
            </View>
            {
              !isMultiLineText
                ?
                <View style={styles.fieldContainer}>
                  <FieldComponent {...neededProps} />
                </View>
                :
                <View style={styles.fieldContainer}>
                  <Text style={{...styles.text, color: neededProps.value && neededProps.value.length ? 'black' : '#c7c7c7'}}>
                    {neededProps.value ? neededProps.value.length : '0'}/480
                  </Text>
                </View>
            }
          </View>
          {isMultiLineText && (
            <View style={{...styles.fieldContainer, marginBottom: 10}}>
              <FieldComponent isMultiLine={true} {...neededProps}  />
            </View>
          )}
        </View>
      )
    }
 
  }
 
  return connect(mapStateToProps, mapDispatchToProps)(InspectionField);
}
 
const mapStateToProps = state => {
  return {
    nextFieldId: state.questionnaire.nextFieldId,
    fieldId: state.questionnaire.fieldId
  }
};
 
const mapDispatchToProps = dispatch => {
  return {
    updateAttribute: (id, value) => dispatch(updateAttribute(id, value)),
    selectNextAttribute: () => dispatch(selectNextAttribute()),
    selectPrevAttribute: () => dispatch(selectPrevAttribute()),
    resetNextAttribute: () => dispatch(resetNextAttribute()),
    selectAttribute: field => dispatch(selectAttribute(field)),
    toggleButtons: () => dispatch(toggleButtons())
  }
};
 
const styles = StyleSheet.create({
  container: {
    flex: 1,
    height: 'auto',
    minHeight: 60,
    display: 'flex',
    flexDirection: 'row'
  },
  fieldContainer: {
    flex: 1,
    display: 'flex',
    justifyContent: 'center',
    alignItems: 'center'
  },
  text: {
    color: 'black',
    fontSize: 16,
    textAlign: 'center',
    fontWeight: '600'
  },
  defaultValue: {
    color: 'grey',
    fontSize: 12,
    textDecorationLine: 'line-through'
  }
});
