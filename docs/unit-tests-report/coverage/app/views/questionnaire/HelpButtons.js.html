import React, { Component } from 'react';
import { StyleSheet, View, Keyboard } from 'react-native';
import { connect } from 'react-redux';
import Icon from 'react-native-vector-icons/MaterialCommunityIcons';
 
import { selectNextAttribute, selectPrevAttribute } from './actions/questionnaire.actions';
 
/**
 * Help buttons component
 */
export class HelpButtons extends Component {
 
  /**
   * Initiate state
   */
  constructor(props) {
    super(props);
    this.state = {
      keyboardHeight: 0
    };
    this.isKeyboardShow = this.isKeyboardShow.bind(this);
  }
 
  /**
   * Add listener for keyboardDidShow event
   */
  componentDidMount() {
    this.keyboardDidShowListener = Keyboard.addListener('keyboardDidShow', this.isKeyboardShow);
  }
 
  /**
   * Remove listener
   */
  componentWillUnmount() {
    this.keyboardDidShowListener.remove();
  }
 
  /**
   * Save height of keyboard
   */
  isKeyboardShow(event) {
    this.setState({keyboardHeight: event.endCoordinates.height})
  };
 
  /**
   * Schedules a help buttons should be rendered
   */
  render() {
    return (
      <View>
        {this.props.showHelpButtons && (
          <View style={{...styles.iconContainer, bottom: this.state.keyboardHeight}}>
            <Icon
              onPress={this.props.selectPrevAttribute}
              size={25}
              color="black"
              name="arrow-up"
            />
            <Icon
              onPress={this.props.selectNextAttribute}
              size={25}
              color="black"
              name="arrow-down"
            />
          </View>
        )}
      </View>
    )
  }
}
 
const styles = StyleSheet.create({
  iconContainer: {
    position: 'absolute',
    backgroundColor: '#7a7a7a',
    opacity: 0.7,
    display: 'flex',
    flexDirection: 'row',
    alignItems: 'center',
    justifyContent: 'space-around',
    height: 45,
    width: 90,
    borderRadius: 5,
    left: 0,
    zIndex: 100
  }
});
 
const mapStateToProps = state => {
  return {
    showHelpButtons: state.questionnaire.showHelpButtons
  };
};
 
const mapDispatchToProps = dispatch => {
  return {
    selectNextAttribute: () => dispatch(selectNextAttribute()),
    selectPrevAttribute: () => dispatch(selectPrevAttribute())
  }
};
 
export default connect(mapStateToProps, mapDispatchToProps)(HelpButtons);
