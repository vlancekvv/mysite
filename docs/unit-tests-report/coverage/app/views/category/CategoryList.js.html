import  React, { Component } from 'react'
import { View, StyleSheet, Text, Dimensions, TouchableOpacity, ScrollView } from 'react-native'
import Icon from 'react-native-vector-icons/MaterialCommunityIcons'
import { connect } from 'react-redux'
import {
  checkTotal,
  triggerLoader,
  changeCurrentPhoto,
  assignCategory,
  validateCategory, toggleAssignPhotoModal
} from '../../actions'
import SubCategoryList from './SubCategoryList'
import Slot from './Slot'
import StorageService from "../../services/StorageService";
import PhotoConfiguration from "./PhotoConfiguration";
 
class CategoryList extends Component {
 
  constructor(props) {
    super(props);
    this.state = {
      collapse: false,
      validate: false,
      modal: false
    }
  }
 
  componentWillUpdate(prevProps) {
    if (prevProps.photosObj.length !== this.props.photosObj.length) {
      this.setState({collapse: true});
    }
  }
 
  toggleModal = () => {
    this.setState({ modal: !this.state.modal })
  };
 
  assignPhotoToCategory = (description, photo, categoryKey) => {
    let item = {
      ...photo,
      category_key: categoryKey || this.props.categoryKey,
      description
    };
    let oldCategory = item.category;
    item.category = this.props.neededCategory;
    StorageService.saveItem(item);
    this.props.assignCategory(item, this.props.neededCategory, oldCategory);
    this.props.validateCategory(item.category);
    if (oldCategory.length) {
      this.props.validateCategory(oldCategory);
    }
    if (categoryKey) {
      this.props.toggleAssignPhotoModal();
    }
  };
 
  getSlots = (category, disable, isReorder) => {
    let slots = [], minPhotos = this.props.attributes.find(attr => attr.name === category).properties.minPhotos;
    for (let index = 0; index < minPhotos; index++) {
      slots.push(<Slot disable = {disable}
                       key={10 + index}
                       index={index + 1}
                       isReorder={isReorder}
                       assignPhotoToCategory={this.assignPhotoToCategory}/>);
    }
    return slots;
  };
 
  _renderSubCategoryPhoto = () => {
    const slot = this.getSlots(this.props.category, true, false);
    this.props.photosObj.forEach((photoObject, key) => {
      if(photoObject.category_key === 'opt' ){
        slot.push(
          <SubCategoryList
            renderSlot ={this._renderSubSlot}
            isReorder={false}
            toggleModal={this.toggleModal}
            handleAssignDescription={this.props.handleAssignDescription}
            deAssignCategory={this.props.deAssignCategory}
            deletePhoto = {this.props.deletePhoto}
            assignPhotoToCategory={this.assignPhotoToCategory}
            key={key}
            disable = {false}
            navigate={this.props.navigate}
            index={photoObject.category_key}
            item={photoObject}/>)
      } else {
        slot.splice(photoObject.category_key - 1, 1,
          <SubCategoryList
            renderSlot={this._renderSubSlot}
            isReorder={false}
            toggleModal={this.toggleModal}
            handleAssignDescription={this.props.handleAssignDescription}
            handleAssignCategory={this.props.handleAssignCategory}
            deAssignCategory={this.props.deAssignCategory}
            deletePhoto={this.props.deletePhoto}
            assignPhotoToCategory={this.assignPhotoToCategory}
            key={key}
            disable={false}
            navigate={this.props.navigate}
            index={photoObject.category_key}
            item={photoObject}/>)
      }
    });
    if (this.props.attributes.find(attr => attr.name === this.props.category).properties.maxPhotos > slot.length) {
      slot.push(<Slot disable={true} key={100} index={"opt"}
                      assignPhotoToCategory={this.assignPhotoToCategory}/>);
    }
    return slot
  };
 
  _renderSubSlot = (isReorder) => {
    const slot = this.getSlots(this.props.neededCategory, false, isReorder);
    this.props.photos.forEach((photoObject, key) => {
      if (photoObject.category_key <= 6 && photoObject.category === this.props.neededCategory) {
        slot.splice(photoObject.category_key - 1, 1,
          <SubCategoryList
            disable={!isReorder}
            isReorder={isReorder}
            renderSlot={this._renderSubSlot}
            toggleModal={this.toggleModal}
            handleAssignDescription={this.props.handleAssignDescription}
            deAssignCategory={this.props.deAssignCategory}
            deletePhoto={this.props.deletePhoto}
            assignPhotoToCategory={this.assignPhotoToCategory}
            key={key}
            navigate={this.props.navigate}
            index={photoObject.category_key}
            item={photoObject}/>)
      }
    });
    return slot;
  };
 
  render() {
    const tablet = this.props.tablet;
    return (
     <View>
      <TouchableOpacity
            style={styles.header}
            onPress={() => {this.setState({collapse: !this.state.collapse})}}>
        <Text style={styles.text}>{this.props.category}</Text>
        <View style={!tablet ? styles.rightContainer : styles.rightContainerTablet}>
          <View style={{
            position: 'relative',
            zIndex: 2,
            width:30,
            height:30,
            borderRadius: 15,
            backgroundColor: this.props.validate || '#FFFFFF',
            justifyContent:'center',
            alignItems: 'center',
          }}>
            <Text style={styles.textTotal}>{this.props.photosObj.length}</Text>
          </View>
              <Icon
                style={styles.icon}
                name={this.state.collapse ? "chevron-down" : 'chevron-up'}
                size={35}
                color="#3c5296"
                marginLeft={10}
                backgroundColor="transparent"/>
        </View>
       </TouchableOpacity>
      { this.state.collapse ?
          <View style={styles.listContainer}>
            <ScrollView>
              {this._renderSubCategoryPhoto()}
            </ScrollView>
          </View>
        : null
      }
       <PhotoConfiguration
         renderSlot={this._renderSubSlot}
         showModal={this.state.modal}
         handleAssignDescription={this.props.handleAssignDescription}
         deAssignCategory={this.props.deAssignCategory}
         deletePhoto={this.props.deletePhoto}
         assignPhotoToCategory={this.assignPhotoToCategory}
         toggleModal={this.toggleModal}
         navigate={this.props.navigate}/>
     </View>
    )
  }
}

const mapStateToProps = state => {
  return {
    photos: state.App.dataPhotos,
    total:state.Home.totalFile,
    tablet: state.App.isTablet,
    neededCategory: state.App.neededCategory,
    modelName: state.App.modelName,
    orders: state.Api.orders,
    order: state.App.currentOrder,
    currentOrderObj: state.App.currentOrderObj,
    attributes: state.App.attributes,
    categoryKey: state.App.categoryKey
  }
};
 
const mapDispatchToProps = dispatch => {
  return {
    checkTotal: (total) => dispatch(checkTotal(total)),
    triggerLoader: () => dispatch(triggerLoader()),
    changeCurrentPhoto : (id) =>dispatch(changeCurrentPhoto(id)),
    assignCategory: (item, category, oldCategory) => dispatch(assignCategory(item, category, oldCategory)),
    validateCategory: category => dispatch(validateCategory(category)),
    toggleAssignPhotoModal: () => dispatch(toggleAssignPhotoModal())
  }
};
 
export default connect(mapStateToProps, mapDispatchToProps)(CategoryList)
